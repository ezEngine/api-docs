36 EZ_FORCE_INLINE ezSimdVec4f ezSimdMath::ACos(const ezSimdVec4f& f)
37 {
38  ezSimdVec4f x1 = f.Abs();
39  ezSimdVec4f x2 = x1.CompMul(x1);
40  ezSimdVec4f x3 = x2.CompMul(x1);
41 
42  ezSimdVec4f s = x1 * -0.2121144f + ezSimdVec4f(1.5707288f);
43  s += x2 * 0.0742610f;
44  s += x3 * -0.0187293f;
45  s = s.CompMul((ezSimdVec4f(1.0f) - x1).GetSqrt());
46 
47  return ezSimdVec4f::Select(f >= ezSimdVec4f::ZeroVector(), s, ezSimdVec4f(ezMath::Pi<float>()) - s);

52 EZ_FORCE_INLINE ezSimdVec4f ezSimdMath::ATan(const ezSimdVec4f& f)
53 {
54  ezSimdVec4f x = f.Abs();
55  ezSimdVec4f t0 = ezSimdVec4f::Select(x < ezSimdVec4f(1.0f), x, x.GetReciprocal());
56  ezSimdVec4f t1 = t0.CompMul(t0);
57  ezSimdVec4f poly = ezSimdVec4f(0.0872929f);
58  poly = ezSimdVec4f(-0.301895f) + poly.CompMul(t1);
59  poly = ezSimdVec4f(1.0f) + poly.CompMul(t1);
60  poly = poly.CompMul(t0);
61  t0 = ezSimdVec4f::Select(x < ezSimdVec4f(1.0f), poly, ezSimdVec4f(ezMath::Pi<float>() * 0.5f) - poly);
62 
63  return ezSimdVec4f::Select(f < ezSimdVec4f::ZeroVector(), -t0, t0);

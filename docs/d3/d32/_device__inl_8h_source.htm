166template <typename HandleType>
167void ezGALDevice::ReviveDeadObject(ezUInt32 uiType, HandleType handle)
168{
169 ezUInt32 uiHandle = handle.GetInternalID().m_Data;
170
171 for (ezUInt32 i = 0; i < m_DeadObjects.GetCount(); ++i)
172 {
173 const auto& deadObject = m_DeadObjects[i];
174
175 if (deadObject.m_uiType == uiType && deadObject.m_uiHandle == uiHandle)
176 {
177 m_DeadObjects.RemoveAtAndCopy(i);
178 return;
179 }
180 }

41  void* ezGuardedAllocation::Allocate(size_t uiSize, size_t uiAlign)
42  {
43  EZ_ASSERT_DEV(ezMath::IsPowerOf2((ezUInt32)uiAlign), "Alignment must be power of two");
44  uiAlign = ezMath::Max<size_t>(uiAlign, EZ_ALIGNMENT_MINIMUM);
45 
46  size_t uiAlignedSize = ezMemoryUtils::AlignSize(uiSize, uiAlign);
47  size_t uiTotalSize = uiAlignedSize + sizeof(AlloctionMetaData);
48 
49  // align to full pages and add one page in front and one in back
50  size_t uiPageSize = m_uiPageSize;
51  size_t uiFullPageSize = ezMemoryUtils::AlignSize(uiTotalSize, uiPageSize);
52  void* pMemory = VirtualAlloc(nullptr, uiFullPageSize + 2 * uiPageSize, MEM_RESERVE, PAGE_NOACCESS);
53  EZ_ASSERT_DEV(pMemory != nullptr, "Could not reserve memory pages. Error Code '{0}'", ezArgErrorCode(::GetLastError()));
54 
55  // add one page and commit the payload pages
56  pMemory = ezMemoryUtils::AddByteOffset(pMemory, uiPageSize);
57  void* ptr = VirtualAlloc(pMemory, uiFullPageSize, MEM_COMMIT, PAGE_READWRITE);
58  EZ_ASSERT_DEV(ptr != nullptr, "Could not commit memory pages. Error Code '{0}'", ezArgErrorCode(::GetLastError()));
59 
60  // store information in meta data
61  AlloctionMetaData* metaData = ezMemoryUtils::AddByteOffset(static_cast<AlloctionMetaData*>(ptr), uiFullPageSize - uiTotalSize);
62  ezMemoryUtils::Construct(metaData, 1);
63  metaData->m_uiSize = uiAlignedSize;
64 
65  // finally add offset to the actual payload
66  ptr = ezMemoryUtils::AddByteOffset(metaData, sizeof(AlloctionMetaData));
67  return ptr;

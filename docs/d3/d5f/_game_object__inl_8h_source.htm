408EZ_ALWAYS_INLINE void ezGameObject::SetGlobalTransform(const ezSimdTransform& transform)
409{
410 UpdateLastGlobalTransform();
411
412 m_pTransformationData->m_globalTransform = transform;
413
414 // ezTransformTemplate<Type>::SetLocalTransform will produce NaNs in w components
415 // of pos and scale if scale.w is not set to 1 here. This only affects builds that
416 // use EZ_SIMD_IMPLEMENTATION_FPU, e.g. arm atm.
417 m_pTransformationData->m_globalTransform.m_Scale.SetW(1.0f);
418 m_pTransformationData->UpdateLocalTransform();
419
420 if (IsStatic())
421 {
422 UpdateGlobalTransformAndBoundsRecursive();
423 }
424}

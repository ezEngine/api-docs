98 virtual ezUInt64 SkipBytes(ezUInt64 uiBytesToSkip)
99 {
100 ezUInt8 uiTempBuffer[1024];
101
102 ezUInt64 uiBytesSkipped = 0;
103
104 while (uiBytesSkipped < uiBytesToSkip)
105 {
106 ezUInt64 uiBytesToRead = ezMath::Min<ezUInt64>(uiBytesToSkip - uiBytesSkipped, 1024);
107
108 ezUInt64 uiBytesRead = ReadBytes(uiTempBuffer, uiBytesToRead);
109
110 uiBytesSkipped += uiBytesRead;
111
112 // Terminate early if the stream didn't read as many bytes as we requested (EOF for example)
113 if (uiBytesRead < uiBytesToRead)
114 break;
115 }
116
117 return uiBytesSkipped;
118 }

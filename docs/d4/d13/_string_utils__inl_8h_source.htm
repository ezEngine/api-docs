81inline ezUInt32 ezStringUtils::GetCharacterCount(const char* szUtf8, const char* pStringEnd)
82{
83 if (IsNullOrEmpty(szUtf8))
84 return 0;
85
86 ezUInt32 uiCharacters = 0;
87
88 while ((szUtf8 < pStringEnd) && (*szUtf8 != '\0'))
89 {
90 // skip all the Utf8 continuation bytes
91 if (!ezUnicodeUtils::IsUtf8ContinuationByte(*szUtf8))
92 ++uiCharacters;

94 ++szUtf8;
95 }

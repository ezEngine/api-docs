62 ezResult ezDirectoryWatcher::OpenDirectory(const ezString& absolutePath, ezBitflags<Watch> whatToWatch)
63 {
64  EZ_ASSERT_DEV(m_sDirectoryPath.IsEmpty(), "Directory already open, call CloseDirectory first!");
65  ezStringBuilder sPath(absolutePath);
66  sPath.MakeCleanPath();
67  sPath.Trim("/");
68 
69  m_pImpl->m_whatToWatch = whatToWatch;
70  m_pImpl->m_filter = FILE_NOTIFY_CHANGE_FILE_NAME;
71  if (whatToWatch.IsSet(Watch::Writes) || whatToWatch.AreAllSet(Watch::Deletes | Watch::Subdirectories))
72  {
73  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_LAST_WRITE;
74  m_pImpl->m_mirror = EZ_DEFAULT_NEW(ezFileSystemMirrorType);
75  m_pImpl->m_mirror->AddDirectory(sPath).AssertSuccess();
76  }
77 
78  if (whatToWatch.IsAnySet(Watch::Deletes | Watch::Creates | Watch::Renames))
79  {
80  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_DIR_NAME;
81  }
82 
83  m_pImpl->m_directoryHandle = CreateFileW(ezDosDevicePath(sPath), FILE_LIST_DIRECTORY, FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
84  nullptr, OPEN_EXISTING, FILE_FLAG_BACKUP_SEMANTICS | FILE_FLAG_OVERLAPPED, nullptr);
85  if (m_pImpl->m_directoryHandle == INVALID_HANDLE_VALUE)
86  {
87  return EZ_FAILURE;
88  }
89 
90  m_pImpl->m_overlappedEvent = CreateEvent(nullptr, TRUE, FALSE, nullptr);
91  if (m_pImpl->m_overlappedEvent == INVALID_HANDLE_VALUE)
92  {
93  return EZ_FAILURE;
94  }
95 
96  m_pImpl->DoRead();
97  m_sDirectoryPath = sPath;
98 
99  return EZ_SUCCESS;
100 }

129 void ezDirectoryWatcher::EnumerateChanges(EnumerateChangesFunction func, ezTime waitUpTo)
130 {
131  EZ_ASSERT_DEV(!m_sDirectoryPath.IsEmpty(), "No directory opened!");
132  while (WaitForSingleObject(m_pImpl->m_overlappedEvent, static_cast<DWORD>(waitUpTo.GetMilliseconds())) == WAIT_OBJECT_0)
133  {
134  waitUpTo = ezTime::Zero(); // only wait on the first call to GetQueuedCompletionStatus
135 
136  DWORD numberOfBytes = 0;
137  GetOverlappedResult(m_pImpl->m_directoryHandle, &m_pImpl->m_overlapped, &numberOfBytes, FALSE);
138 
139  // Copy the buffer
140  ezHybridArray<ezUInt8, 4096> buffer;
141  buffer.SetCountUninitialized(numberOfBytes);
142  buffer.GetArrayPtr().CopyFrom(m_pImpl->m_buffer.GetArrayPtr().GetSubArray(0, numberOfBytes));
143 
144  // Reissue the read request
145  m_pImpl->DoRead();
146 
147  if (numberOfBytes == 0)
148  {
149  return;
150  }
151 
152  const ezBitflags<ezDirectoryWatcher::Watch> whatToWatch = m_pImpl->m_whatToWatch;
153 
154  ezFileSystemMirrorType* mirror = m_pImpl->m_mirror.Borrow();
155 
156  MoveEvent lastMoveFrom;
157 
158  // Progress the messages
159  auto info = (const FILE_NOTIFY_EXTENDED_INFORMATION*)buffer.GetData();
160  while (true)
161  {
162  auto directory = ezArrayPtr<const WCHAR>(info->FileName, info->FileNameLength / sizeof(WCHAR));
163  int bytesNeeded = WideCharToMultiByte(CP_UTF8, 0, directory.GetPtr(), directory.GetCount(), nullptr, 0, nullptr, nullptr);
164  if (bytesNeeded > 0)
165  {
166  ezHybridArray<char, 1024> dir;
167  dir.SetCountUninitialized(bytesNeeded);
168  WideCharToMultiByte(CP_UTF8, 0, directory.GetPtr(), directory.GetCount(), dir.GetData(), dir.GetCount(), nullptr, nullptr);
169  ezDirectoryWatcherAction action = ezDirectoryWatcherAction::None;
170  bool fireEvent = false;
171 
172  ezStringBuilder eventFilePath = m_sDirectoryPath;
173  eventFilePath.AppendPath(ezStringView(dir.GetData(), dir.GetCount()));
174  eventFilePath.MakeCleanPath();
175 
176  if ((info->FileAttributes & FILE_ATTRIBUTE_DIRECTORY) == 0)
177  {
178  switch (info->Action)
179  {
180  case FILE_ACTION_ADDED:
181  DEBUG_LOG("FILE_ACTION_ADDED {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
182  action = ezDirectoryWatcherAction::Added;
183  fireEvent = whatToWatch.IsSet(ezDirectoryWatcher::Watch::Creates);
184  if (mirror)
185  {
186  bool fileAlreadyExists = false;
187  mirror->AddFile(eventFilePath.GetData(), true, &fileAlreadyExists, nullptr).AssertSuccess();
188  if (fileAlreadyExists)
189  {
190  fireEvent = false;
191  }
192  }
193  break;
194  case FILE_ACTION_REMOVED:
195  DEBUG_LOG("FILE_ACTION_REMOVED {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
196  action = ezDirectoryWatcherAction::Removed;
197  fireEvent = whatToWatch.IsSet(ezDirectoryWatcher::Watch::Deletes);
198  if (mirror)
199  {
200  mirror->RemoveFile(eventFilePath.GetData()).AssertSuccess();
201  }
202  break;
203  case FILE_ACTION_MODIFIED:
204  {
205  DEBUG_LOG("FILE_ACTION_MODIFIED {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
206  action = ezDirectoryWatcherAction::Modified;
207  fireEvent = whatToWatch.IsAnySet(ezDirectoryWatcher::Watch::Writes);
208  bool fileAreadyKnown = false;
209  bool addPending = false;
210  if (mirror)
211  {
212  mirror->AddFile(eventFilePath.GetData(), false, &fileAreadyKnown, &addPending).AssertSuccess();
213  }
214  if (fileAreadyKnown && addPending)
215  {
216  fireEvent = false;
217  }
218  }
219  break;
220  case FILE_ACTION_RENAMED_OLD_NAME:
221  DEBUG_LOG("FILE_ACTION_RENAMED_OLD_NAME {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
222  EZ_ASSERT_DEV(lastMoveFrom.IsEmpty(), "there should be no pending move from");
223  action = ezDirectoryWatcherAction::RenamedOldName;
224  fireEvent = whatToWatch.IsAnySet(ezDirectoryWatcher::Watch::Renames);
225  EZ_ASSERT_DEV(lastMoveFrom.IsEmpty(), "there should be no pending last move from");
226  lastMoveFrom = {eventFilePath, false};
227  break;
228  case FILE_ACTION_RENAMED_NEW_NAME:
229  DEBUG_LOG("FILE_ACTION_RENAMED_NEW_NAME {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
230  action = ezDirectoryWatcherAction::RenamedNewName;
231  fireEvent = whatToWatch.IsAnySet(ezDirectoryWatcher::Watch::Renames);
232  EZ_ASSERT_DEV(!lastMoveFrom.IsEmpty() && !lastMoveFrom.isDirectory, "last move from doesn't match");
233  if (mirror)
234  {
235  mirror->RemoveFile(lastMoveFrom.path).AssertSuccess();
236  mirror->AddFile(eventFilePath, false, nullptr, nullptr).AssertSuccess();
237  }
238  lastMoveFrom.Clear();
239  break;
240  }
241 
242  if (fireEvent)
243  {
244  func(eventFilePath, action, ezDirectoryWatcherType::File);
245  }
246  }
247  else
248  {
249  switch (info->Action)
250  {
251  case FILE_ACTION_ADDED:
252  {
253  DEBUG_LOG("DIR_ACTION_ADDED {}", eventFilePath);
254  bool directoryAlreadyKnown = false;
255  if (mirror)
256  {
257  mirror->AddDirectory(eventFilePath, &directoryAlreadyKnown).AssertSuccess();
258  }
259 
260  if (whatToWatch.IsSet(Watch::Creates) && !directoryAlreadyKnown)
261  {
262  func(eventFilePath, ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::Directory);
263  }
264 
265  // Whenever we add a directory we might be "to late" to see changes inside it.
266  // So iterate the file system and make sure we track all files / subdirectories
267  ezFileSystemIterator subdirIt;
268 
269  subdirIt.StartSearch(eventFilePath.GetData(),
270  whatToWatch.IsSet(ezDirectoryWatcher::Watch::Subdirectories)
271  ? ezFileSystemIteratorFlags::ReportFilesAndFoldersRecursive
272  : ezFileSystemIteratorFlags::ReportFiles);
273 
274  ezStringBuilder tmpPath2;
275  for (; subdirIt.IsValid(); subdirIt.Next())
276  {
277  const ezFileStats& stats = subdirIt.GetStats();
278  stats.GetFullPath(tmpPath2);
279  if (stats.m_bIsDirectory)
280  {
281  directoryAlreadyKnown = false;
282  if (mirror)
283  {
284  mirror->AddDirectory(tmpPath2, &directoryAlreadyKnown).AssertSuccess();
285  }
286  if (whatToWatch.IsSet(ezDirectoryWatcher::Watch::Creates) && !directoryAlreadyKnown)
287  {
288  func(tmpPath2, ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::Directory);
289  }
290  }
291  else
292  {
293  bool fileExistsAlready = false;
294  if (mirror)
295  {
296  mirror->AddFile(tmpPath2, false, &fileExistsAlready, nullptr).AssertSuccess();
297  }
298  if (whatToWatch.IsSet(ezDirectoryWatcher::Watch::Creates) && !fileExistsAlready)
299  {
300  func(tmpPath2, ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::File);
301  }
302  }
303  }
304  }
305  break;
306  case FILE_ACTION_REMOVED:
307  DEBUG_LOG("DIR_ACTION_REMOVED {}", eventFilePath);
308  if (whatToWatch.IsSet(Watch::Deletes))
309  {
310  if (mirror && whatToWatch.IsSet(Watch::Subdirectories))
311  {
312  mirror->Enumerate(eventFilePath, [&](const ezStringBuilder& path, ezFileSystemMirrorType::Type type) {
313  func(path, ezDirectoryWatcherAction::Removed, (type == ezFileSystemMirrorType::Type::File) ? ezDirectoryWatcherType::File : ezDirectoryWatcherType::Directory);
314  })
315  .AssertSuccess();
316  }
317  func(eventFilePath, ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::Directory);
318  }
319  if (mirror)
320  {
321  mirror->RemoveDirectory(eventFilePath).AssertSuccess();
322  }
323  break;
324  case FILE_ACTION_RENAMED_OLD_NAME:
325  DEBUG_LOG("DIR_ACTION_OLD_NAME {}", eventFilePath);
326  EZ_ASSERT_DEV(lastMoveFrom.IsEmpty(), "there should be no pending move from");
327  lastMoveFrom = {eventFilePath, true};
328  break;
329  case FILE_ACTION_RENAMED_NEW_NAME:
330  DEBUG_LOG("DIR_ACTION_NEW_NAME {}", eventFilePath);
331  EZ_ASSERT_DEV(!lastMoveFrom.IsEmpty(), "rename old name and rename new name should always appear in pairs");
332  if (mirror)
333  {
334  mirror->MoveDirectory(lastMoveFrom.path, eventFilePath).AssertSuccess();
335  }
336  if (whatToWatch.IsSet(Watch::Renames))
337  {
338  func(lastMoveFrom.path, ezDirectoryWatcherAction::RenamedOldName, ezDirectoryWatcherType::Directory);
339  func(eventFilePath, ezDirectoryWatcherAction::RenamedNewName, ezDirectoryWatcherType::Directory);
340  }
341  lastMoveFrom.Clear();
342  break;
343  default:
344  break;
345  }
346  }
347  }
348  if (info->NextEntryOffset == 0)
349  break;
350  else
351  info = (const FILE_NOTIFY_EXTENDED_INFORMATION*)(((ezUInt8*)info) + info->NextEntryOffset);
352  }
353  }
354 }

357 void ezDirectoryWatcher::EnumerateChanges(ezArrayPtr<ezDirectoryWatcher*> watchers, EnumerateChangesFunction func, ezTime waitUpTo)
358 {
359  ezHybridArray<HANDLE, 16> events;
360  events.SetCount(watchers.GetCount());
361 
362  for (ezUInt32 i = 0; i < watchers.GetCount(); ++i)
363  {
364  events[i] = watchers[i]->m_pImpl->m_overlappedEvent;
365  }
366 
367  // Wait for any of the watchers to have some data ready
368  if (WaitForMultipleObjects(events.GetCount(), events.GetData(), FALSE, static_cast<DWORD>(waitUpTo.GetMilliseconds())) == WAIT_TIMEOUT)
369  {
370  return;
371  }
372 
373  // Iterate all of them to make sure we report all changes up to this point.
374  for (ezDirectoryWatcher* watcher : watchers)
375  {
376  watcher->EnumerateChanges(func);
377  }

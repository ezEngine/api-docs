30 ezResult ezDirectoryWatcher::OpenDirectory(const ezString& absolutePath, ezBitflags<Watch> whatToWatch)
31 {
32  EZ_ASSERT_DEV(m_sDirectoryPath.IsEmpty(), "Directory already open, call CloseDirectory first!");
33  ezStringBuilder sPath(absolutePath);
34  sPath.MakeCleanPath();
35  sPath.Trim("/");
36 
37  m_pImpl->m_watchSubdirs = whatToWatch.IsSet(Watch::Subdirectories);
38  m_pImpl->m_filter = 0;
39  if (whatToWatch.IsSet(Watch::Reads))
40  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_LAST_ACCESS;
41  if (whatToWatch.IsSet(Watch::Writes))
42  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_LAST_WRITE;
43  if (whatToWatch.IsSet(Watch::Creates))
44  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_CREATION;
45  if (whatToWatch.IsSet(Watch::Renames))
46  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_FILE_NAME | FILE_NOTIFY_CHANGE_DIR_NAME;
47 
48  m_pImpl->m_directoryHandle = CreateFileW(ezDosDevicePath(sPath), FILE_LIST_DIRECTORY, FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
49  nullptr, OPEN_EXISTING, FILE_FLAG_BACKUP_SEMANTICS | FILE_FLAG_OVERLAPPED, nullptr);
50  if (m_pImpl->m_directoryHandle == INVALID_HANDLE_VALUE)
51  {
52  return EZ_FAILURE;
53  }
54 
55  m_pImpl->m_completionPort = CreateIoCompletionPort(m_pImpl->m_directoryHandle, nullptr, 0, 1);
56  if (m_pImpl->m_completionPort == INVALID_HANDLE_VALUE)
57  {

92 void ezDirectoryWatcher::EnumerateChanges(EnumerateChangesFunction func)
93 {
94  EZ_ASSERT_DEV(!m_sDirectoryPath.IsEmpty(), "No directory opened!");
95  OVERLAPPED* lpOverlapped;
96  DWORD numberOfBytes;
97  ULONG_PTR completionKey;
98  while (GetQueuedCompletionStatus(m_pImpl->m_completionPort, &numberOfBytes, &completionKey, &lpOverlapped, 0) != 0)
99  {
100  if (numberOfBytes <= 0)
101  {
102  ezLog::Debug("GetQueuedCompletionStatus with size 0. Should not happen according to msdn.");
103  m_pImpl->DoRead();
104  continue;
105  }
106  // Copy the buffer
107 
108  ezHybridArray<ezUInt8, 4096> buffer;
109  buffer.SetCountUninitialized(numberOfBytes);
110  buffer.GetArrayPtr().CopyFrom(m_pImpl->m_buffer.GetArrayPtr().GetSubArray(0, numberOfBytes));
111 
112  // Reissue the read request
113  m_pImpl->DoRead();
114 
115  // Progress the messages
116  auto info = (const FILE_NOTIFY_INFORMATION*)buffer.GetData();
117  while (true)
118  {
119  auto directory = ezArrayPtr<const WCHAR>(info->FileName, info->FileNameLength / sizeof(WCHAR));
120  int bytesNeeded = WideCharToMultiByte(CP_UTF8, 0, directory.GetPtr(), directory.GetCount(), nullptr, 0, nullptr, nullptr);
121  if (bytesNeeded > 0)
122  {
123  ezHybridArray<char, 1024> dir;
124  dir.SetCountUninitialized(bytesNeeded + 1);
125  WideCharToMultiByte(CP_UTF8, 0, directory.GetPtr(), directory.GetCount(), dir.GetData(), dir.GetCount(), nullptr, nullptr);
126  dir[bytesNeeded] = '\0';
127  ezDirectoryWatcherAction action = ezDirectoryWatcherAction::None;
128  switch (info->Action)
129  {
130  case FILE_ACTION_ADDED:
131  action = ezDirectoryWatcherAction::Added;
132  break;
133  case FILE_ACTION_REMOVED:
134  action = ezDirectoryWatcherAction::Removed;
135  break;
136  case FILE_ACTION_MODIFIED:
137  action = ezDirectoryWatcherAction::Modified;
138  break;
139  case FILE_ACTION_RENAMED_OLD_NAME:
140  action = ezDirectoryWatcherAction::RenamedOldName;
141  break;
142  case FILE_ACTION_RENAMED_NEW_NAME:
143  action = ezDirectoryWatcherAction::RenamedNewName;
144  break;
145  }
146 
147  ezStringBuilder cleanDir = dir.GetData();
148  cleanDir.MakeCleanPath();
149 
150  func(cleanDir, action);
151  }
152  if (info->NextEntryOffset == 0)
153  break;
154  else
155  info = (const FILE_NOTIFY_INFORMATION*)(((ezUInt8*)info) + info->NextEntryOffset);
156  }
157  }
158 
159  if (lpOverlapped != nullptr)
160  {
161  EZ_ASSERT_DEV(false, "GetQueuedCompletionStatus returned false but lpOverlapped is not null");
162  }
163 
164  EZ_ASSERT_DEV(GetLastError() == WAIT_TIMEOUT, "GetQueuedCompletionStatus gave an error");
165 }

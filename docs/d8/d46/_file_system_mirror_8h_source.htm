7 // A general problem when implementing a directory watcher is, that moving a folder out of the watched directory only communicates
8 // Which folder was moved (but not to where, nor its contents). This means when a folder is moved out of view,
9 // this needs to be treated as a delete. At the point of the move, it is no longer possible to query the contents of the folder.
10 // So a in memory copy of the file system is required in order to correctly implement a directory watcher.

84 template <typename T>
85 ezResult ezFileSystemMirror<T>::AddDirectory(const char* path, bool* outDirectoryExistsAlready)
86 {
87  ezStringBuilder currentDirAbsPath = path;
88  currentDirAbsPath.MakeCleanPath();
89  EnsureTrailingSlash(currentDirAbsPath);
90 
91  if (m_topLevelDirPath.IsEmpty())
92  {
93  m_topLevelDirPath = currentDirAbsPath;
94  currentDirAbsPath.Shrink(0, 1); // remove trailing /
95 
96  DirEntry* currentDir = &m_topLevelDir;
97 
98  ezHybridArray<DirEntry*, 16> m_dirStack;
99 
100  ezFileSystemIterator files;
101  files.StartSearch(currentDirAbsPath.GetData(), ezFileSystemIteratorFlags::ReportFilesAndFoldersRecursive);
102  for (; files.IsValid(); files.Next())
103  {
104  const ezFileStats& stats = files.GetStats();
105 
106  // In case we are done with a directory, move back up
107  while (currentDirAbsPath != stats.m_sParentPath)
108  {
109  EZ_ASSERT_DEV(m_dirStack.GetCount() > 0, "Unexpected file iteration order");
110  currentDir = m_dirStack.PeekBack();
111  m_dirStack.PopBack();
112  currentDirAbsPath.PathParentDirectory();
113  RemoveTrailingSlash(currentDirAbsPath);
114  }
115 
116  if (stats.m_bIsDirectory)
117  {
118  m_dirStack.PushBack(currentDir);
119  ezStringBuilder subdirName = stats.m_sName;
120  EnsureTrailingSlash(subdirName);
121  auto insertIt = currentDir->m_subDirectories.Insert(subdirName, DirEntry());
122  currentDir = &insertIt.Value();
123  currentDirAbsPath.AppendPath(stats.m_sName);
124  }
125  else
126  {
127  currentDir->m_files.Insert(std::move(stats.m_sName), T{});
128  }
129  }
130  if (outDirectoryExistsAlready != nullptr)
131  {
132  *outDirectoryExistsAlready = false;
133  }
134  }
135  else
136  {
137  DirEntry* parentDir = FindDirectory(currentDirAbsPath);
138  if (parentDir == nullptr)
139  {
140  return EZ_FAILURE;
141  }
142 
143  if (outDirectoryExistsAlready != nullptr)
144  {
145  *outDirectoryExistsAlready = currentDirAbsPath.IsEmpty();
146  }
147 
148  while (!currentDirAbsPath.IsEmpty())
149  {
150  const char* dirEnd = currentDirAbsPath.FindSubString("/");
151  ezStringView subdirName(currentDirAbsPath.GetData(), dirEnd + 1);
152  auto insertIt = parentDir->m_subDirectories.Insert(subdirName, DirEntry());
153  parentDir = &insertIt.Value();
154  currentDirAbsPath.Shrink(ezStringUtils::GetCharacterCount(subdirName.GetStartPointer(), subdirName.GetEndPointer()), 0);
155  }
156  }
157 
158  return EZ_SUCCESS;
159 }

161 template <typename T>
162 ezResult ezFileSystemMirror<T>::AddFile(const char* path, const T& value, bool* outFileExistsAlready, T* outOldValue)
163 {
164  ezStringBuilder sPath = path;
165  DirEntry* dir = FindDirectory(sPath);
166  if (dir == nullptr)
167  {
168  return EZ_FAILURE; // file not under top level directory
169  }
170 
171  const char* szSlashPos = sPath.FindSubString("/");
172 
173  while (szSlashPos != nullptr)
174  {
175  ezStringView subdirName(sPath.GetData(), szSlashPos + 1);
176  auto insertIt = dir->m_subDirectories.Insert(subdirName, DirEntry());
177  dir = &insertIt.Value();
178  sPath.Shrink(ezStringUtils::GetCharacterCount(subdirName.GetStartPointer(), subdirName.GetEndPointer()), 0);
179  szSlashPos = sPath.FindSubString("/");
180  }
181 
182  auto it = dir->m_files.Find(sPath);
183  // Do not add the file twice
184  if (!it.IsValid())
185  {
186  dir->m_files.Insert(sPath, value);
187  if (outFileExistsAlready != nullptr)
188  {
189  *outFileExistsAlready = false;
190  }
191  }
192  else
193  {
194  if (outFileExistsAlready != nullptr)
195  {
196  *outFileExistsAlready = true;
197  }
198  if (outOldValue != nullptr)
199  {
200  *outOldValue = it.Value();
201  }
202  it.Value() = value;
203  }
204  return EZ_SUCCESS;
205 }

262 ezResult ezFileSystemMirror<T>::MoveDirectory(const char* fromPath, const char* toPath)
263 {
264  ezStringBuilder sFromPath = fromPath;
265  ezStringBuilder sFromName = fromPath;
266  sFromPath.PathParentDirectory();
267  EnsureTrailingSlash(sFromPath);
268  sFromName.Shrink(sFromPath.GetCharacterCount(), 0);
269  EnsureTrailingSlash(sFromName);
270 
271 
272  ezStringBuilder sToPath = toPath;
273  ezStringBuilder sToName = toPath;
274  sToPath.PathParentDirectory();
275  EnsureTrailingSlash(sToPath);
276  sToName.Shrink(sToPath.GetCharacterCount(), 0);
277  EnsureTrailingSlash(sToName);
278 
279  DirEntry* moveFromDir = FindDirectory(sFromPath);
280  if (!moveFromDir)
281  {
282  return EZ_FAILURE;
283  }
284  EZ_ASSERT_DEV(sFromPath.IsEmpty(), "move from directory should fully exist");
285 
286  DirEntry* moveToDir = FindDirectory(sToPath);
287  if (!moveToDir)
288  {
289  return EZ_FAILURE;
290  }
291 
292  if (!sToPath.IsEmpty())
293  {
294  do
295  {
296  const char* dirEnd = sToPath.FindSubString("/");
297  ezStringView subdirName(sToPath.GetData(), dirEnd + 1);
298  auto insertIt = moveToDir->m_subDirectories.Insert(subdirName, DirEntry());
299  moveToDir = &insertIt.Value();
300  sToPath.Shrink(0, ezStringUtils::GetCharacterCount(subdirName.GetStartPointer(), subdirName.GetEndPointer()));
301  } while (!sToPath.IsEmpty());
302  }
303 
304  DirEntry movedDir;
305  {
306  auto fromIt = moveFromDir->m_subDirectories.Find(sFromName);
307  if (!fromIt.IsValid())
308  {
309  return EZ_FAILURE;
310  }
311 
312  movedDir = std::move(fromIt.Value());
313  moveFromDir->m_subDirectories.Remove(fromIt);
314  }
315 
316  moveToDir->m_subDirectories.Insert(sToName, std::move(movedDir));
317 
318  return EZ_SUCCESS;
319 }

331 template <typename T>
332 ezResult ezFileSystemMirror<T>::Enumerate(const char* path, EnumerateFunc callbackFunc)
333 {
334  ezHybridArray<ezDirEnumerateState<T>, 16> dirStack;
335  ezStringBuilder sPath = path;
336  if (!sPath.EndsWith("/"))
337  {
338  sPath.Append("/");
339  }
340  DirEntry* dirToEnumerate = FindDirectory(sPath);
341  if (dirToEnumerate == nullptr)
342  {
343  return EZ_FAILURE;
344  }
345  if (!sPath.IsEmpty())
346  {
347  return EZ_FAILURE; // requested folder to enumerate doesn't exist
348  }
349  DirEntry* currentDir = dirToEnumerate;
350  typename ezMap<ezString, ezFileSystemMirror::DirEntry>::Iterator currentSubDirIt = currentDir->m_subDirectories.GetIterator();
351  sPath = path;
352 
353  while (currentDir != nullptr)
354  {
355  if (currentSubDirIt.IsValid())
356  {
357  DirEntry* nextDir = &currentSubDirIt.Value();
358  sPath.AppendPath(currentSubDirIt.Key());
359  currentSubDirIt.Next();
360  dirStack.PushBack({currentDir, currentSubDirIt});
361  currentDir = nextDir;
362  }
363  else
364  {
365  ezStringBuilder sFilePath;
366  for (auto& file : currentDir->m_files)
367  {
368  sFilePath = sPath;
369  sFilePath.AppendPath(file.Key());
370  callbackFunc(sFilePath, Type::File);
371  }
372 
373  if (currentDir != dirToEnumerate)
374  {
375  if (sPath.EndsWith("/") && sPath.GetElementCount() > 1)
376  {
377  sPath.Shrink(0, 1);
378  }
379  callbackFunc(sPath, Type::Directory);
380  }
381 
382  if (dirStack.IsEmpty())
383  {
384  currentDir = nullptr;
385  }
386  else
387  {
388  currentDir = dirStack.PeekBack().dir;
389  currentSubDirIt = dirStack.PeekBack().subDirIt;
390  dirStack.PopBack();
391  sPath.PathParentDirectory();
392  if (sPath.GetElementCount() > 1 && sPath.EndsWith("/"))
393  {
394  sPath.Shrink(0, 1);
395  }
396  }
397  }
398  }
399 
400  return EZ_SUCCESS;
401 }

404 typename ezFileSystemMirror<T>::DirEntry* ezFileSystemMirror<T>::FindDirectory(ezStringBuilder& path)
405 {
406  if (!path.StartsWith(m_topLevelDirPath))
407  {
408  return nullptr;
409  }
410  path.TrimWordStart(m_topLevelDirPath);
411 
412  DirEntry* currentDir = &m_topLevelDir;
413 
414  bool found = false;
415  do
416  {
417  found = false;
418  for (auto& dir : currentDir->m_subDirectories)
419  {
420  if (path.StartsWith(dir.Key()))
421  {
422  currentDir = &dir.Value();
423  path.TrimWordStart(dir.Key());
424  path.TrimWordStart("/");
425  found = true;
426  break;
427  }
428  }
429  } while (found);

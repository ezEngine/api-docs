7// A general problem when implementing a directory watcher is, that moving a folder out of the watched directory only communicates
8// Which folder was moved (but not to where, nor its contents). This means when a folder is moved out of view,
9// this needs to be treated as a delete. At the point of the move, it is no longer possible to query the contents of the folder.
10// So a in memory copy of the file system is required in order to correctly implement a directory watcher.

86template <typename T>
87ezResult ezFileSystemMirror<T>::AddDirectory(ezStringView sPath, bool* out_pDirectoryExistsAlready)
88{
89 ezStringBuilder currentDirAbsPath = sPath;
90 currentDirAbsPath.MakeCleanPath();
91 EnsureTrailingSlash(currentDirAbsPath);
92
93 if (m_sTopLevelDirPath.IsEmpty())
94 {
95 m_sTopLevelDirPath = currentDirAbsPath;
96 currentDirAbsPath.Shrink(0, 1); // remove trailing /
97
98 DirEntry* currentDir = &m_TopLevelDir;
99
100 ezHybridArray<DirEntry*, 16> m_dirStack;
101
102 ezFileSystemIterator files;
103 files.StartSearch(currentDirAbsPath.GetData(), ezFileSystemIteratorFlags::ReportFilesAndFoldersRecursive);
104 for (; files.IsValid(); files.Next())
105 {
106 const ezFileStats& stats = files.GetStats();
107
108 // In case we are done with a directory, move back up
109 while (currentDirAbsPath != stats.m_sParentPath)
110 {
111 EZ_ASSERT_DEV(m_dirStack.GetCount() > 0, "Unexpected file iteration order");
112 currentDir = m_dirStack.PeekBack();
113 m_dirStack.PopBack();
114 currentDirAbsPath.PathParentDirectory();
115 RemoveTrailingSlash(currentDirAbsPath);
116 }
117
118 if (stats.m_bIsDirectory)
119 {
120 m_dirStack.PushBack(currentDir);
121 ezStringBuilder subdirName = stats.m_sName;
122 EnsureTrailingSlash(subdirName);
123 auto insertIt = currentDir->m_subDirectories.Insert(subdirName, DirEntry());
124 currentDir = &insertIt.Value();
125 currentDirAbsPath.AppendPath(stats.m_sName);
126 }
127 else
128 {
129 currentDir->m_files.Insert(std::move(stats.m_sName), T{});
130 }
131 }
132 if (out_pDirectoryExistsAlready != nullptr)
133 {
134 *out_pDirectoryExistsAlready = false;
135 }
136 }
137 else
138 {
139 DirEntry* parentDir = FindDirectory(currentDirAbsPath);
140 if (parentDir == nullptr)
141 {
142 return EZ_FAILURE;
143 }
144
145 if (out_pDirectoryExistsAlready != nullptr)
146 {
147 *out_pDirectoryExistsAlready = currentDirAbsPath.IsEmpty();
148 }
149
150 while (!currentDirAbsPath.IsEmpty())
151 {
152 const char* dirEnd = currentDirAbsPath.FindSubString("/");
153 ezStringView subdirName(currentDirAbsPath.GetData(), dirEnd + 1);
154 auto insertIt = parentDir->m_subDirectories.Insert(subdirName, DirEntry());
155 parentDir = &insertIt.Value();
156 currentDirAbsPath.Shrink(ezStringUtils::GetCharacterCount(subdirName.GetStartPointer(), subdirName.GetEndPointer()), 0);
157 }
158 }
159
160 return EZ_SUCCESS;
161}

163template <typename T>
164ezResult ezFileSystemMirror<T>::AddFile(ezStringView sPath0, const T& value, bool* out_pFileExistsAlready, T* out_pOldValue)
165{
166 ezStringBuilder sPath = sPath0;
167 DirEntry* dir = FindDirectory(sPath);
168 if (dir == nullptr)
169 {
170 return EZ_FAILURE; // file not under top level directory
171 }
172
173 const char* szSlashPos = sPath.FindSubString("/");
174
175 while (szSlashPos != nullptr)
176 {
177 ezStringView subdirName(sPath.GetData(), szSlashPos + 1);
178 auto insertIt = dir->m_subDirectories.Insert(subdirName, DirEntry());
179 dir = &insertIt.Value();
180 sPath.Shrink(ezStringUtils::GetCharacterCount(subdirName.GetStartPointer(), subdirName.GetEndPointer()), 0);
181 szSlashPos = sPath.FindSubString("/");
182 }
183
184 auto it = dir->m_files.Find(sPath);
185 // Do not add the file twice
186 if (!it.IsValid())
187 {
188 dir->m_files.Insert(sPath, value);
189 if (out_pFileExistsAlready != nullptr)
190 {
191 *out_pFileExistsAlready = false;
192 }
193 }
194 else
195 {
196 if (out_pFileExistsAlready != nullptr)
197 {
198 *out_pFileExistsAlready = true;
199 }
200 if (out_pOldValue != nullptr)
201 {
202 *out_pOldValue = it.Value();
203 }
204 it.Value() = value;
205 }
206 return EZ_SUCCESS;
207}

210ezResult ezFileSystemMirror<T>::RemoveFile(ezStringView sPath0)
211{
212 ezStringBuilder sPath = sPath0;
213 DirEntry* dir = FindDirectory(sPath);
214 if (dir == nullptr)
215 {
216 return EZ_FAILURE; // file not under top level directory
217 }
218
219 if (sPath.FindSubString("/") != nullptr)
220 {
221 return EZ_FAILURE; // file does not exist
222 }
223
224 if (dir->m_files.GetCount() == 0)
225 {
226 return EZ_FAILURE; // there are no files in this directory
227 }
228
229 auto it = dir->m_files.Find(sPath);
230 if (!it.IsValid())
231 {
232 return EZ_FAILURE; // file does not exist
233 }
234
235 dir->m_files.Remove(it);
236 return EZ_SUCCESS;

240ezResult ezFileSystemMirror<T>::RemoveDirectory(ezStringView sPath)
241{
242 ezStringBuilder parentPath = sPath;
243 ezStringBuilder dirName = sPath;
244 parentPath.PathParentDirectory();
245 EnsureTrailingSlash(parentPath);
246 dirName.Shrink(parentPath.GetCharacterCount(), 0);
247 EnsureTrailingSlash(dirName);
248
249 DirEntry* parentDir = FindDirectory(parentPath);
250 if (parentDir == nullptr || !parentPath.IsEmpty())
251 {
252 return EZ_FAILURE;
253 }
254
255 if (!parentDir->m_subDirectories.Remove(dirName))
256 {
257 return EZ_FAILURE;
258 }
259
260 return EZ_SUCCESS;

263template <typename T>
264ezResult ezFileSystemMirror<T>::MoveDirectory(ezStringView sFromPath0, ezStringView sToPath0)
265{
266 ezStringBuilder sFromPath = sFromPath0;
267 ezStringBuilder sFromName = sFromPath0;
268 sFromPath.PathParentDirectory();
269 EnsureTrailingSlash(sFromPath);
270 sFromName.Shrink(sFromPath.GetCharacterCount(), 0);
271 EnsureTrailingSlash(sFromName);
272
273
274 ezStringBuilder sToPath = sToPath0;
275 ezStringBuilder sToName = sToPath0;
276 sToPath.PathParentDirectory();
277 EnsureTrailingSlash(sToPath);
278 sToName.Shrink(sToPath.GetCharacterCount(), 0);
279 EnsureTrailingSlash(sToName);
280
281 DirEntry* moveFromDir = FindDirectory(sFromPath);
282 if (!moveFromDir)
283 {
284 return EZ_FAILURE;
285 }
286 EZ_ASSERT_DEV(sFromPath.IsEmpty(), "move from directory should fully exist");
287
288 DirEntry* moveToDir = FindDirectory(sToPath);
289 if (!moveToDir)
290 {
291 return EZ_FAILURE;
292 }
293
294 if (!sToPath.IsEmpty())
295 {
296 do
297 {
298 const char* dirEnd = sToPath.FindSubString("/");
299 ezStringView subdirName(sToPath.GetData(), dirEnd + 1);
300 auto insertIt = moveToDir->m_subDirectories.Insert(subdirName, DirEntry());
301 moveToDir = &insertIt.Value();
302 sToPath.Shrink(0, ezStringUtils::GetCharacterCount(subdirName.GetStartPointer(), subdirName.GetEndPointer()));
303 } while (!sToPath.IsEmpty());
304 }
305
306 DirEntry movedDir;
307 {
308 auto fromIt = moveFromDir->m_subDirectories.Find(sFromName);
309 if (!fromIt.IsValid())
310 {
311 return EZ_FAILURE;
312 }
313
314 movedDir = std::move(fromIt.Value());
315 moveFromDir->m_subDirectories.Remove(fromIt);
316 }
317
318 moveToDir->m_subDirectories.Insert(sToName, std::move(movedDir));
319
320 return EZ_SUCCESS;
321}

333template <typename T>
334ezResult ezFileSystemMirror<T>::Enumerate(ezStringView sPath0, EnumerateFunc callbackFunc)
335{
336 ezHybridArray<ezDirEnumerateState<T>, 16> dirStack;
337 ezStringBuilder sPath = sPath0;
338 if (!sPath.EndsWith("/"))
339 {
340 sPath.Append("/");
341 }
342 DirEntry* dirToEnumerate = FindDirectory(sPath);
343 if (dirToEnumerate == nullptr)
344 {
345 return EZ_FAILURE;
346 }
347 if (!sPath.IsEmpty())
348 {
349 return EZ_FAILURE; // requested folder to enumerate doesn't exist
350 }
351 DirEntry* currentDir = dirToEnumerate;
352 typename ezMap<ezString, ezFileSystemMirror::DirEntry>::Iterator currentSubDirIt = currentDir->m_subDirectories.GetIterator();
353 sPath = sPath0;
354
355 while (currentDir != nullptr)
356 {
357 if (currentSubDirIt.IsValid())
358 {
359 DirEntry* nextDir = &currentSubDirIt.Value();
360 sPath.AppendPath(currentSubDirIt.Key());
361 currentSubDirIt.Next();
362 dirStack.PushBack({currentDir, currentSubDirIt});
363 currentDir = nextDir;
364 }
365 else
366 {
367 ezStringBuilder sFilePath;
368 for (auto& file : currentDir->m_files)
369 {
370 sFilePath = sPath;
371 sFilePath.AppendPath(file.Key());
372 callbackFunc(sFilePath, Type::File);
373 }
374
375 if (currentDir != dirToEnumerate)
376 {
377 if (sPath.EndsWith("/") && sPath.GetElementCount() > 1)
378 {
379 sPath.Shrink(0, 1);
380 }
381 callbackFunc(sPath, Type::Directory);
382 }
383
384 if (dirStack.IsEmpty())
385 {
386 currentDir = nullptr;
387 }
388 else
389 {
390 currentDir = dirStack.PeekBack().dir;
391 currentSubDirIt = dirStack.PeekBack().subDirIt;
392 dirStack.PopBack();
393 sPath.PathParentDirectory();
394 if (sPath.GetElementCount() > 1 && sPath.EndsWith("/"))
395 {
396 sPath.Shrink(0, 1);
397 }
398 }
399 }
400 }
401
402 return EZ_SUCCESS;
403}

407{
408 ezStringBuilder sPath = sPath0;
409 DirEntry* dir = FindDirectory(sPath);
410 if (dir == nullptr)
411 {
412 return EZ_FAILURE; // file not under top level directory

414

432template <typename T>
433typename ezFileSystemMirror<T>::DirEntry* ezFileSystemMirror<T>::FindDirectory(ezStringBuilder& path)
434{
435 if (!path.StartsWith(m_sTopLevelDirPath))
436 {
437 return nullptr;
438 }
439 path.TrimWordStart(m_sTopLevelDirPath);
440
441 DirEntry* currentDir = &m_TopLevelDir;
442
443 bool found = false;
444 do
445 {
446 found = false;
447 for (auto& dir : currentDir->m_subDirectories)
448 {
449 if (path.StartsWith(dir.Key()))
450 {
451 currentDir = &dir.Value();
452 path.TrimWordStart(dir.Key());
453 path.TrimWordStart("/");
454 found = true;
455 break;
456 }
457 }
458 } while (found);
459
460 return currentDir;

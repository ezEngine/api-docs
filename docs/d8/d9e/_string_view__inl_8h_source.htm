133 void ezStringView::Split(bool bReturnEmptyStrings, Container& ref_output, const char* szSeparator1, const char* szSeparator2 /*= nullptr*/, const char* szSeparator3 /*= nullptr*/, const char* szSeparator4 /*= nullptr*/, const char* szSeparator5 /*= nullptr*/, const char* szSeparator6 /*= nullptr*/) const
134 {
135  ref_output.Clear();
136 
137  if (IsEmpty())
138  return;
139 
140  const ezUInt32 uiParams = 6;
141 
142  const ezStringView seps[uiParams] = {szSeparator1, szSeparator2, szSeparator3, szSeparator4, szSeparator5, szSeparator6};
143 
144  const char* szReadPos = GetStartPointer();
145 
146  while (true)
147  {
148  const char* szFoundPos = ezUnicodeUtils::GetMaxStringEnd<char>();
149  ezUInt32 uiFoundSeparator = 0;
150 
151  for (ezUInt32 i = 0; i < uiParams; ++i)
152  {
153  const char* szFound = ezStringUtils::FindSubString(szReadPos, seps[i].GetStartPointer(), GetEndPointer(), seps[i].GetEndPointer());
154 
155  if ((szFound != nullptr) && (szFound < szFoundPos))
156  {
157  szFoundPos = szFound;
158  uiFoundSeparator = i;
159  }
160  }
161 
162  // nothing found
163  if (szFoundPos == ezUnicodeUtils::GetMaxStringEnd<char>())
164  {
165  const ezUInt32 uiLen = ezStringUtils::GetStringElementCount(szReadPos, GetEndPointer());
166 
167  if (bReturnEmptyStrings || (uiLen > 0))
168  ref_output.PushBack(ezStringView(szReadPos, szReadPos + uiLen));
169 
170  return;
171  }
172 
173  if (bReturnEmptyStrings || (szFoundPos > szReadPos))
174  ref_output.PushBack(ezStringView(szReadPos, szFoundPos));
175 
176  szReadPos = szFoundPos + seps[uiFoundSeparator].GetElementCount();
177  }
178 }

93  static ezBitflags<ezPropertyFlags> GetParameterFlags()
94  {
95  using CleanType = typename ezTypeTraits<Type>::NonConstReferencePointerType;
96  ezBitflags<ezPropertyFlags> flags;
97  constexpr ezVariantType::Enum type = static_cast<ezVariantType::Enum>(ezVariantTypeDeduction<CleanType>::value);
98  if constexpr (std::is_same<CleanType, ezVariant>::value ||
99  std::is_same<Type, const char*>::value || // We treat const char* as a basic type and not a pointer.
100  (type >= ezVariantType::FirstStandardType && type <= ezVariantType::LastStandardType))
101  flags.Add(ezPropertyFlags::StandardType);
102  else if constexpr (ezIsEnum<CleanType>::value)
103  flags.Add(ezPropertyFlags::IsEnum);
104  else if constexpr (ezIsBitflags<CleanType>::value)
105  flags.Add(ezPropertyFlags::Bitflags);
106  else
107  flags.Add(ezPropertyFlags::Class);
108 
109  if constexpr (std::is_const<typename ezTypeTraits<Type>::NonReferencePointerType>::value)
110  flags.Add(ezPropertyFlags::Const);
111 
112  if constexpr (std::is_pointer<Type>::value && !std::is_same<Type, const char*>::value)
113  flags.Add(ezPropertyFlags::Pointer);
114 
115  if constexpr (std::is_reference<Type>::value)
116  flags.Add(ezPropertyFlags::Reference);
117 
118  return flags;
119  }

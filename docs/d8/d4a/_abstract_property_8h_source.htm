94 template <class Type>
95 static ezBitflags<ezPropertyFlags> GetParameterFlags()
96 {
97 using CleanType = typename ezTypeTraits<Type>::NonConstReferencePointerType;
98 ezBitflags<ezPropertyFlags> flags;
99 constexpr ezVariantType::Enum type = static_cast<ezVariantType::Enum>(ezVariantTypeDeduction<CleanType>::value);
100 if constexpr (std::is_same<CleanType, ezVariant>::value ||
101 std::is_same<Type, const char*>::value || // We treat const char* as a basic type and not a pointer.
102 (type >= ezVariantType::FirstStandardType && type <= ezVariantType::LastStandardType))
103 flags.Add(ezPropertyFlags::StandardType);
104 else if constexpr (ezIsEnum<CleanType>::value)
105 flags.Add(ezPropertyFlags::IsEnum);
106 else if constexpr (ezIsBitflags<CleanType>::value)
107 flags.Add(ezPropertyFlags::Bitflags);
108 else
109 flags.Add(ezPropertyFlags::Class);
110
111 if constexpr (std::is_const<typename ezTypeTraits<Type>::NonReferencePointerType>::value)
112 flags.Add(ezPropertyFlags::Const);
113
114 if constexpr (std::is_pointer<Type>::value && !std::is_same<Type, const char*>::value)
115 flags.Add(ezPropertyFlags::Pointer);
116
117 if constexpr (std::is_reference<Type>::value)
118 flags.Add(ezPropertyFlags::Reference);
119
120 return flags;
121 }
122};

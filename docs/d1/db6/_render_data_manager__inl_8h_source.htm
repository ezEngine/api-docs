30EZ_FORCE_INLINE void ezRenderDataManager::FillPerInstanceData(ezPerInstanceData& out_perInstanceData, const ezGameObject* pObject, const ezTransform& globalTransform, ezUInt32 uiUniqueID /*= 0*/, const ezColor& color /*= ezColor::White*/, const ezVec4& vCustomData /*= ezVec4(0, 1, 0, 1)*/, float fBoundingSphereRadius /*= 1.0f*/, ezUInt32 uiRandomSeed /*= 0*/)
31{
32 ezMat4 objectToWorld = globalTransform.GetAsMat4();
33 out_perInstanceData.ObjectToWorld = objectToWorld;
34
35 if (globalTransform.HasOnlyUniformScaling())
36 {
37 out_perInstanceData.ObjectToWorldNormal = objectToWorld;
38 }
39 else
40 {
41 ezMat3 mInverse = objectToWorld.GetRotationalPart();
42 mInverse.Invert(0.0f).IgnoreResult();
43 // we explicitly ignore the return value here (success / failure)
44 // because when we have a scale of 0 (which happens temporarily during editing) that would be annoying
45
46 out_perInstanceData.ObjectToWorldNormal = mInverse.GetTranspose();
47 }
48
49 if (pObject != nullptr)
50 {
51 out_perInstanceData.BoundingSphereRadius = pObject->GetGlobalBounds().m_fSphereRadius;
52 out_perInstanceData.RandomSeed = pObject->GetStableRandomSeed();
53 }
54 else
55 {
56 out_perInstanceData.BoundingSphereRadius = fBoundingSphereRadius;
57 out_perInstanceData.RandomSeed = uiRandomSeed;
58 }
59
60 out_perInstanceData.GameObjectID = uiUniqueID;
61 out_perInstanceData.Reserved = 0;
62
63 out_perInstanceData.Color = color;
64 out_perInstanceData.CustomData = vCustomData;
65}

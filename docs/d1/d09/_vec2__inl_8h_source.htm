106EZ_IMPLEMENT_IF_FLOAT_TYPE inline ezResult ezVec2Template<Type>::NormalizeIfNotZero(const ezVec2Template<Type>& vFallback, Type fEpsilon)
107{
108 EZ_NAN_ASSERT(&vFallback);
109
110 const Type fLength = GetLength();
111
112 if (!ezMath::IsFinite(fLength) || ezMath::IsZero(fLength, fEpsilon))
113 {
114 *this = vFallback;
115 return EZ_FAILURE;
116 }
117
118 *this /= fLength;
119 return EZ_SUCCESS;
120}

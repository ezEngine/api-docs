183template <typename Type>
184inline void ezAngleTemplate<Type>::NormalizeRange()
185{
186 constexpr Type fTwoPi = Type(2.0) * Pi();
187 constexpr Type fTwoPiTen = Type(10.0) * Pi();
188
189 if (m_fRadian > fTwoPiTen || m_fRadian < -fTwoPiTen)
190 {
191 m_fRadian = ezMath::Mod(m_fRadian, fTwoPi);
192 }
193
194 while (m_fRadian >= fTwoPi)
195 {
196 m_fRadian -= fTwoPi;
197 }
198
199 while (m_fRadian < Type(0.0))
200 {
201 m_fRadian += fTwoPi;
202 }
203}

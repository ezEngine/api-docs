87inline void ezSimdBBox::ExpandToInclude(const ezSimdVec4f* pPoints, ezUInt32 uiNumPoints, ezUInt32 uiStride)
88{
89 EZ_ASSERT_DEBUG(pPoints != nullptr, "Array may not be nullptr.");
90 EZ_ASSERT_DEBUG(uiStride >= sizeof(ezSimdVec4f), "Data may not overlap.");
91
92 const ezSimdVec4f* pCur = pPoints;
93
94 for (ezUInt32 i = 0; i < uiNumPoints; ++i)

96 ExpandToInclude(*pCur);

161EZ_ALWAYS_INLINE void ezSimdBBox::Transform(const ezSimdMat4f& mMat)
162{
163 const ezSimdVec4f center = GetCenter();
164 const ezSimdVec4f halfExtents = center - m_Min;
165
166 const ezSimdVec4f newCenter = mMat.TransformPosition(center);
167
168 ezSimdVec4f newHalfExtents = mMat.m_col0.Abs() * halfExtents.x();
169 newHalfExtents += mMat.m_col1.Abs() * halfExtents.y();
170 newHalfExtents += mMat.m_col2.Abs() * halfExtents.z();
171
172 *this = ezSimdBBox::MakeFromCenterAndHalfExtents(newCenter, newHalfExtents);
173}

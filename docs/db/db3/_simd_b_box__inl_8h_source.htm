135 EZ_ALWAYS_INLINE void ezSimdBBox::Transform(const ezSimdMat4f& mat)
136 {
137  const ezSimdVec4f center = GetCenter();
138  const ezSimdVec4f halfExtents = center - m_Min;
139 
140  const ezSimdVec4f newCenter = mat.TransformPosition(center);
141 
142  ezSimdVec4f newHalfExtents = mat.m_col0.Abs() * halfExtents.x();
143  newHalfExtents += mat.m_col1.Abs() * halfExtents.y();
144  newHalfExtents += mat.m_col2.Abs() * halfExtents.z();
145 
146  SetCenterAndHalfExtents(newCenter, newHalfExtents);
147 }

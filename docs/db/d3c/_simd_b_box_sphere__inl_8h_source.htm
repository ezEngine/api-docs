100 {
101  ezSimdFloat radius = m_CenterAndRadius.w();
102  m_CenterAndRadius = mat.TransformPosition(m_CenterAndRadius);
103 
104  ezSimdFloat maxRadius = mat.m_col0.Dot<3>(mat.m_col0);
105  maxRadius = maxRadius.Max(mat.m_col1.Dot<3>(mat.m_col1));
106  maxRadius = maxRadius.Max(mat.m_col2.Dot<3>(mat.m_col2));
107  radius *= maxRadius.GetSqrt();
108 
109  m_CenterAndRadius.SetW(radius);
110 
111  ezSimdVec4f newHalfExtents = mat.m_col0.Abs() * m_BoxHalfExtents.x();
112  newHalfExtents += mat.m_col1.Abs() * m_BoxHalfExtents.y();
113  newHalfExtents += mat.m_col2.Abs() * m_BoxHalfExtents.z();
114 
115  m_BoxHalfExtents = newHalfExtents.CompMin(ezSimdVec4f(radius));

15 {
16  int junk;
17  int mib[4];
18  struct kinfo_proc info;
19  size_t size;
20 
21  // Initialize the flags so that, if sysctl fails for some bizarre
22  // reason, we get a predictable result.
23  info.kp_proc.p_flag = 0;
24 
25  // Initialize mib, which tells sysctl the info we want, in this case
26  // we're looking for information about a specific process ID.
27  mib[0] = CTL_KERN;
28  mib[1] = KERN_PROC;
29  mib[2] = KERN_PROC_PID;
30  mib[3] = getpid();
31 
32  // Call sysctl.
33  size = sizeof(info);
34  junk = sysctl(mib, sizeof(mib) / sizeof(*mib), &info, &size, NULL, 0);
35  assert(junk == 0);
36 
37  // We're being debugged if the P_TRACED flag is set.
38  return ((info.kp_proc.p_flag & P_TRACED) != 0);
39 }

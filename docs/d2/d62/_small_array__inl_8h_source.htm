31template <typename T, ezUInt16 Size>
32void ezSmallArrayBase<T, Size>::CopyFrom(const ezArrayPtr<const T>& other, ezAllocator* pAllocator)
33{
34 EZ_ASSERT_DEV(other.GetCount() <= ezSmallInvalidIndex, "Can't copy {} elements to small array. Maximum count is {}", other.GetCount(), ezSmallInvalidIndex);
35
36 if (GetData() == other.GetPtr())
37 {
38 if (m_uiCount == other.GetCount())
39 return;
40
41 EZ_ASSERT_DEV(m_uiCount > other.GetCount(), "Dangling array pointer. The given array pointer points to invalid memory.");
42 T* pElements = GetElementsPtr();
43 ezMemoryUtils::Destruct(pElements + other.GetCount(), m_uiCount - other.GetCount());
44 m_uiCount = static_cast<ezUInt16>(other.GetCount());
45 return;
46 }
47
48 const ezUInt32 uiOldCount = m_uiCount;
49 const ezUInt32 uiNewCount = other.GetCount();
50
51 if (uiNewCount > uiOldCount)
52 {
53 Reserve(static_cast<ezUInt16>(uiNewCount), pAllocator);
54 T* pElements = GetElementsPtr();
55 ezMemoryUtils::Copy(pElements, other.GetPtr(), uiOldCount);
56 ezMemoryUtils::CopyConstructArray(pElements + uiOldCount, other.GetPtr() + uiOldCount, uiNewCount - uiOldCount);
57 }
58 else
59 {
60 T* pElements = GetElementsPtr();
61 ezMemoryUtils::Copy(pElements, other.GetPtr(), uiNewCount);
62 ezMemoryUtils::Destruct(pElements + uiNewCount, uiOldCount - uiNewCount);
63 }
64
65 m_uiCount = static_cast<ezUInt16>(uiNewCount);
66}

68template <typename T, ezUInt16 Size>
69void ezSmallArrayBase<T, Size>::MoveFrom(ezSmallArrayBase<T, Size>&& other, ezAllocator* pAllocator)
70{
71 Clear();
72
73 if (other.m_uiCapacity > Size)
74 {
75 if (m_uiCapacity > Size)
76 {
77 // only delete our own external storage
78 EZ_DELETE_RAW_BUFFER(pAllocator, m_pElements);
79 }
80
81 m_uiCapacity = other.m_uiCapacity;
82 m_pElements = other.m_pElements;
83 }
84 else
85 {
86 ezMemoryUtils::RelocateConstruct(GetElementsPtr(), other.GetElementsPtr(), other.m_uiCount);
87 }
88
89 m_uiCount = other.m_uiCount;
90 m_uiUserData = other.m_uiUserData;
91
92 // reset the other array to not reference the data anymore
93 other.m_pElements = nullptr;
94 other.m_uiCount = 0;
95 other.m_uiCapacity = 0;
96}

292template <typename T, ezUInt16 Size>
293void ezSmallArrayBase<T, Size>::RemoveAtAndCopy(ezUInt32 uiIndex, ezUInt16 uiNumElements /*= 1*/)
294{
295 EZ_ASSERT_DEV(uiIndex + uiNumElements <= m_uiCount, "Out of bounds access. Array has {0} elements, trying to remove element at index {1}.", m_uiCount, uiIndex + uiNumElements - 1);
296
297 T* pElements = GetElementsPtr();

299 m_uiCount -= uiNumElements;

303template <typename T, ezUInt16 Size>
304void ezSmallArrayBase<T, Size>::RemoveAtAndSwap(ezUInt32 uiIndex, ezUInt16 uiNumElements /*= 1*/)
305{
306 EZ_ASSERT_DEV(uiIndex + uiNumElements <= m_uiCount, "Out of bounds access. Array has {0} elements, trying to remove element at index {1}.", m_uiCount, uiIndex + uiNumElements - 1);
307
308 T* pElements = GetElementsPtr();
309
310 for (ezUInt32 i = 0; i < uiNumElements; ++i)
311 {
312 m_uiCount--;
313
314 if (m_uiCount != uiIndex)
315 {
316 pElements[uiIndex] = std::move(pElements[m_uiCount]);
317 }
318 ezMemoryUtils::Destruct(pElements + m_uiCount, 1);
319 ++uiIndex;
320 }
321}

497template <typename T, ezUInt16 Size>
498void ezSmallArrayBase<T, Size>::Reserve(ezUInt16 uiCapacity, ezAllocator* pAllocator)
499{
500 if (m_uiCapacity >= uiCapacity)
501 return;
502
503 const ezUInt32 uiCurCap = static_cast<ezUInt32>(m_uiCapacity);
504 ezUInt32 uiNewCapacity = uiCurCap + (uiCurCap / 2);
505
506 uiNewCapacity = ezMath::Max<ezUInt32>(uiNewCapacity, uiCapacity);
507 uiNewCapacity = ezMemoryUtils::AlignSize<ezUInt32>(uiNewCapacity, CAPACITY_ALIGNMENT);
508 uiNewCapacity = ezMath::Min<ezUInt32>(uiNewCapacity, 0xFFFFu);
509
510 SetCapacity(static_cast<ezUInt16>(uiNewCapacity), pAllocator);
511}

513template <typename T, ezUInt16 Size>
514void ezSmallArrayBase<T, Size>::Compact(ezAllocator* pAllocator)
515{
516 if (IsEmpty())
517 {
518 if (m_uiCapacity > Size)
519 {
520 // completely deallocate all data, if the array is empty.
521 EZ_DELETE_RAW_BUFFER(pAllocator, m_pElements);
522 }
523
524 m_uiCapacity = Size;
525 m_pElements = nullptr;
526 }
527 else if (m_uiCapacity > Size)
528 {
529 ezUInt32 uiNewCapacity = ezMemoryUtils::AlignSize<ezUInt32>(m_uiCount, CAPACITY_ALIGNMENT);
530 uiNewCapacity = ezMath::Min<ezUInt32>(uiNewCapacity, 0xFFFFu);
531
532 if (m_uiCapacity != uiNewCapacity)
533 SetCapacity(static_cast<ezUInt16>(uiNewCapacity), pAllocator);
534 }
535}

559template <typename T, ezUInt16 Size>
560void ezSmallArrayBase<T, Size>::SetCapacity(ezUInt16 uiCapacity, ezAllocator* pAllocator)
561{
562 if (m_uiCapacity > Size && uiCapacity > m_uiCapacity)
563 {
564 m_pElements = EZ_EXTEND_RAW_BUFFER(pAllocator, m_pElements, m_uiCount, uiCapacity);
565 m_uiCapacity = uiCapacity;
566 }
567 else
568 {
569 // special case when migrating from in-place to external storage or shrinking
570 T* pOldElements = GetElementsPtr();
571
572 const ezUInt32 uiOldCapacity = m_uiCapacity;
573 const ezUInt32 uiNewCapacity = uiCapacity;
574 m_uiCapacity = ezMath::Max(uiCapacity, Size);
575
576 if (uiNewCapacity > Size)
577 {
578 // new external storage
579 T* pNewElements = EZ_NEW_RAW_BUFFER(pAllocator, T, uiCapacity);
580 ezMemoryUtils::RelocateConstruct(pNewElements, pOldElements, m_uiCount);
581 m_pElements = pNewElements;
582 }
583 else
584 {
585 // Re-use inplace storage
586 ezMemoryUtils::RelocateConstruct(GetElementsPtr(), pOldElements, m_uiCount);
587 }
588
589 if (uiOldCapacity > Size)
590 {
591 EZ_DELETE_RAW_BUFFER(pAllocator, pOldElements);
592 }
593 }
594}

31inline ezSimdBSphere ezSimdBSphere::MakeFromPoints(const ezSimdVec4f* pPoints, ezUInt32 uiNumPoints, ezUInt32 uiStride /*= sizeof(ezSimdVec4f)*/)
32{
33 EZ_ASSERT_DEBUG(pPoints != nullptr, "The array must not be empty.");
34 EZ_ASSERT_DEBUG(uiStride >= sizeof(ezSimdVec4f), "The data must not overlap.");
35 EZ_ASSERT_DEBUG(uiNumPoints > 0, "The array must contain at least one point.");
36
37 ezSimdBSphere res;
38
39 const ezSimdVec4f* pCur = pPoints;
40
41 ezSimdVec4f vCenter = ezSimdVec4f::MakeZero();
42 for (ezUInt32 i = 0; i < uiNumPoints; ++i)
43 {
44 vCenter += *pCur;
45 pCur = ezMemoryUtils::AddByteOffset(pCur, uiStride);
46 }
47
48 res.m_CenterAndRadius = vCenter / ezSimdFloat(uiNumPoints);
49
50 pCur = pPoints;
51
52 ezSimdFloat fMaxDistSquare = ezSimdFloat::MakeZero();
53 for (ezUInt32 i = 0; i < uiNumPoints; ++i)
54 {
55 const ezSimdFloat fDistSQR = (*pCur - res.m_CenterAndRadius).GetLengthSquared<3>();
56 fMaxDistSquare = fMaxDistSquare.Max(fDistSQR);
57
58 pCur = ezMemoryUtils::AddByteOffset(pCur, uiStride);
59 }
60
61 res.m_CenterAndRadius.SetW(fMaxDistSquare.GetSqrt());
62
63 return res;
64}

141 m_CenterAndRadius = mMat.TransformPosition(m_CenterAndRadius);

237 template <typename Type>
238 void ezQuatTemplate<Type>::SetFromMat3(const ezMat3Template<Type>& m)
239 {
240  EZ_NAN_ASSERT(&m);
241 
242  const Type trace = m.Element(0, 0) + m.Element(1, 1) + m.Element(2, 2);
243  const Type half = (Type)0.5;
244 
245  Type val[4];
246 
247  if (trace > (Type)0)
248  {
249  Type s = ezMath::Sqrt(trace + (Type)1);
250  Type t = half / s;
251 
252  val[0] = (m.Element(1, 2) - m.Element(2, 1)) * t;
253  val[1] = (m.Element(2, 0) - m.Element(0, 2)) * t;
254  val[2] = (m.Element(0, 1) - m.Element(1, 0)) * t;
255 
256  val[3] = half * s;
257  }
258  else
259  {
260  const ezInt32 next[] = {1, 2, 0};
261  ezInt32 i = 0;
262 
263  if (m.Element(1, 1) > m.Element(0, 0))
264  i = 1;
265 
266  if (m.Element(2, 2) > m.Element(i, i))
267  i = 2;
268 
269  ezInt32 j = next[i];
270  ezInt32 k = next[j];
271 
272  Type s = ezMath::Sqrt(m.Element(i, i) - (m.Element(j, j) + m.Element(k, k)) + (Type)1);
273  Type t = half / s;
274 
275  val[i] = half * s;
276  val[3] = (m.Element(j, k) - m.Element(k, j)) * t;
277  val[j] = (m.Element(i, j) + m.Element(j, i)) * t;
278  val[k] = (m.Element(i, k) + m.Element(k, i)) * t;
279  }
280 
281  v.x = val[0];
282  v.y = val[1];
283  v.z = val[2];
284  w = val[3];
285 }

293 template <typename Type>
294 void ezQuatTemplate<Type>::SetShortestRotation(const ezVec3Template<Type>& vDirFrom, const ezVec3Template<Type>& vDirTo)
295 {
296  const ezVec3Template<Type> v0 = vDirFrom.GetNormalized();
297  const ezVec3Template<Type> v1 = vDirTo.GetNormalized();
298 
299  const Type fDot = v0.Dot(v1);
300 
301  // if both vectors are identical -> no rotation needed
302  if (ezMath::IsEqual(fDot, (Type)1, (Type)0.0001))
303  {
304  SetIdentity();
305  return;
306  }
307  else
308  // if both vectors are opposing
309  if (ezMath::IsEqual(fDot, (Type)-1, (Type)0.0001))
310  {
311  // find an axis, that is not identical and not opposing, ezVec3Template::Cross-product to find perpendicular vector, rotate around that
312  if (ezMath::Abs(v0.Dot(ezVec3Template<Type>(1, 0, 0))) < (Type)0.8)
313  SetFromAxisAndAngle(v0.CrossRH(ezVec3Template<Type>(1, 0, 0)).GetNormalized(), ezAngle::Radian(ezMath::Pi<float>()));
314  else
315  SetFromAxisAndAngle(v0.CrossRH(ezVec3Template<Type>(0, 1, 0)).GetNormalized(), ezAngle::Radian(ezMath::Pi<float>()));
316 
317  return;
318  }
319 
320  const ezVec3Template<Type> c = v0.CrossRH(v1);
321  const Type d = v0.Dot(v1);
322  const Type s = ezMath::Sqrt(((Type)1 + d) * (Type)2);
323 
324  EZ_ASSERT_DEBUG(c.IsValid(), "SetShortestRotation failed.");
325 
326  v = c / s;
327  w = s / (Type)2;
328 
329  Normalize();
330 }

332 template <typename Type>
333 void ezQuatTemplate<Type>::SetSlerp(const ezQuatTemplate<Type>& qFrom, const ezQuatTemplate<Type>& qTo, Type t)
334 {
335  EZ_ASSERT_DEBUG((t >= (Type)0) && (t <= (Type)1), "Invalid lerp factor.");
336 
337  const Type one = 1;
338  const Type qdelta = (Type)1 - (Type)0.001;
339 
340  const Type fDot = (qFrom.v.x * qTo.v.x + qFrom.v.y * qTo.v.y + qFrom.v.z * qTo.v.z + qFrom.w * qTo.w);
341 
342  Type cosTheta = fDot;
343 
344  bool bFlipSign = false;
345  if (cosTheta < (Type)0)
346  {
347  bFlipSign = true;
348  cosTheta = -cosTheta;
349  }
350 
351  Type t0, t1;
352 
353  if (cosTheta < qdelta)
354  {
355  ezAngle theta = ezMath::ACos(cosTheta);
356 
357  // use sqrtInv(1+c^2) instead of 1.0/sin(theta)
358  const Type iSinTheta = (Type)1 / ezMath::Sqrt(one - (cosTheta * cosTheta));
359  const ezAngle tTheta = t * theta;
360 
361  Type s0 = ezMath::Sin(theta - tTheta);
362  Type s1 = ezMath::Sin(tTheta);
363 
364  t0 = s0 * iSinTheta;
365  t1 = s1 * iSinTheta;
366  }
367  else
368  {
369  // If q0 is nearly the same as q1 we just linearly interpolate
370  t0 = one - t;
371  t1 = t;
372  }
373 
374  if (bFlipSign)
375  t1 = -t1;
376 
377  v.x = t0 * qFrom.v.x;
378  v.y = t0 * qFrom.v.y;
379  v.z = t0 * qFrom.v.z;
380  w = t0 * qFrom.w;
381 
382  v.x += t1 * qTo.v.x;
383  v.y += t1 * qTo.v.y;
384  v.z += t1 * qTo.v.z;
385  w += t1 * qTo.w;
386 
387  Normalize();
388 }

402 template <typename Type>
403 void ezQuatTemplate<Type>::GetAsEulerAngles(ezAngle& out_x, ezAngle& out_y, ezAngle& out_z) const
404 {
406 
409  auto& yaw = out_z;
410  auto& pitch = out_y;
411  auto& roll = out_x;
412  // roll (x-axis rotation)
413  const double sinr = 2.0 * (w * v.x + v.y * v.z);
414  const double cosr = 1.0 - 2.0 * (v.x * v.x + v.y * v.y);
415  roll = ezMath::ATan2((float)sinr, (float)cosr);
416 
417  // pitch (y-axis rotation)
418  const double sinp = 2.0 * (w * v.y - v.z * v.x);
419  if (ezMath::Abs(sinp) >= 1.0)
420  pitch = ezAngle::Radian(copysign(ezMath::Pi<float>() / 2.0f, (float)sinp)); // use 90 degrees if out of range
421  else
422  pitch = ezMath::ASin((float)sinp);
423 
424  // yaw (z-axis rotation)
425  const double siny = 2.0 * (w * v.z + v.x * v.y);
426  const double cosy = 1.0 - 2.0 * (v.y * v.y + v.z * v.z);
427  yaw = ezMath::ATan2((float)siny, (float)cosy);
428 }

434

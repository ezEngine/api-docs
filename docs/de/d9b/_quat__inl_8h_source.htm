145{

147

274template <typename Type>
275ezQuatTemplate<Type> ezQuatTemplate<Type>::MakeFromMat3(const ezMat3Template<Type>& m)
276{
277 EZ_NAN_ASSERT(&m);
278
279 const Type trace = m.Element(0, 0) + m.Element(1, 1) + m.Element(2, 2);
280 const Type half = (Type)0.5;
281
282 Type val[4];
283
284 if (trace > (Type)0)
285 {
286 Type s = ezMath::Sqrt(trace + (Type)1);
287 Type t = half / s;
288
289 val[0] = (m.Element(1, 2) - m.Element(2, 1)) * t;
290 val[1] = (m.Element(2, 0) - m.Element(0, 2)) * t;
291 val[2] = (m.Element(0, 1) - m.Element(1, 0)) * t;
292
293 val[3] = half * s;
294 }
295 else
296 {
297 const ezInt32 next[] = {1, 2, 0};
298 ezInt32 i = 0;
299
300 if (m.Element(1, 1) > m.Element(0, 0))
301 i = 1;
302
303 if (m.Element(2, 2) > m.Element(i, i))
304 i = 2;
305
306 ezInt32 j = next[i];
307 ezInt32 k = next[j];
308
309 Type s = ezMath::Sqrt(m.Element(i, i) - (m.Element(j, j) + m.Element(k, k)) + (Type)1);
310 Type t = half / s;
311
312 val[i] = half * s;
313 val[3] = (m.Element(j, k) - m.Element(k, j)) * t;
314 val[j] = (m.Element(i, j) + m.Element(j, i)) * t;
315 val[k] = (m.Element(i, k) + m.Element(k, i)) * t;
316 }
317
318 ezQuatTemplate<Type> q;
319 q.x = val[0];
320 q.y = val[1];
321 q.z = val[2];
322 q.w = val[3];
323 return q;
324}

362template <typename Type>
363ezQuatTemplate<Type> ezQuatTemplate<Type>::MakeShortestRotation(const ezVec3Template<Type>& vDirFrom, const ezVec3Template<Type>& vDirTo)
364{
365 const ezVec3Template<Type> v0 = vDirFrom.GetNormalized();
366 const ezVec3Template<Type> v1 = vDirTo.GetNormalized();
367
368 const Type fDot = v0.Dot(v1);
369
370 // if both vectors are identical -> no rotation needed
371 if (ezMath::IsEqual(fDot, (Type)1, (Type)0.0000001))
372 {
373 return MakeIdentity();
374 }
375 else if (ezMath::IsEqual(fDot, (Type)-1, (Type)0.0000001)) // if both vectors are opposing
376 {
377 // find an axis, that is not identical and not opposing, ezVec3Template::Cross-product to find perpendicular vector, rotate around that
378 if (ezMath::Abs(v0.Dot(ezVec3Template<Type>(1, 0, 0))) < (Type)0.8)
379 return MakeFromAxisAndAngle(v0.CrossRH(ezVec3Template<Type>(1, 0, 0)).GetNormalized(), ezAngle::MakeFromRadian(ezMath::Pi<float>()));
380 else
381 return MakeFromAxisAndAngle(v0.CrossRH(ezVec3Template<Type>(0, 1, 0)).GetNormalized(), ezAngle::MakeFromRadian(ezMath::Pi<float>()));
382 }
383
384 const ezVec3Template<Type> c = v0.CrossRH(v1);
385 const Type d = v0.Dot(v1);
386 const Type s = ezMath::Sqrt(((Type)1 + d) * (Type)2);
387
388 EZ_ASSERT_DEBUG(c.IsValid(), "SetShortestRotation failed.");
389
390 const Type fOneDivS = 1.0f / s;
391
392 ezQuatTemplate<Type> q;
393 q.x = c.x * fOneDivS;
394 q.y = c.y * fOneDivS;
395 q.z = c.z * fOneDivS;
396 q.w = s / (Type)2;
397 q.Normalize();
398
399 return q;
400}

402template <typename Type>
403ezQuatTemplate<Type> ezQuatTemplate<Type>::MakeSlerp(const ezQuatTemplate<Type>& qFrom, const ezQuatTemplate<Type>& qTo, Type t)
404{
405 EZ_ASSERT_DEBUG((t >= (Type)0) && (t <= (Type)1), "Invalid lerp factor.");
406
407 const Type one = 1;
408 const Type qdelta = (Type)1 - (Type)0.001;
409
410 const Type fDot = (qFrom.x * qTo.x + qFrom.y * qTo.y + qFrom.z * qTo.z + qFrom.w * qTo.w);
411
412 Type cosTheta = fDot;
413
414 bool bFlipSign = false;
415 if (cosTheta < (Type)0)
416 {
417 bFlipSign = true;
418 cosTheta = -cosTheta;
419 }
420
421 Type t0, t1;
422
423 if (cosTheta < qdelta)
424 {
425 ezAngle theta = ezMath::ACos((float)cosTheta);
426
427 // use sqrtInv(1+c^2) instead of 1.0/sin(theta)
428 const Type iSinTheta = (Type)1 / ezMath::Sqrt(one - (cosTheta * cosTheta));
429 const ezAngle tTheta = static_cast<float>(t) * theta;
430
431 Type s0 = ezMath::Sin(theta - tTheta);
432 Type s1 = ezMath::Sin(tTheta);
433
434 t0 = s0 * iSinTheta;
435 t1 = s1 * iSinTheta;
436 }
437 else
438 {
439 // If q0 is nearly the same as q1 we just linearly interpolate
440 t0 = one - t;
441 t1 = t;
442 }
443
444 if (bFlipSign)
445 t1 = -t1;
446
447 ezQuatTemplate<Type> q;
448
449 q.x = t0 * qFrom.x;
450 q.y = t0 * qFrom.y;
451 q.z = t0 * qFrom.z;
452 q.w = t0 * qFrom.w;
453
454 q.x += t1 * qTo.x;
455 q.y += t1 * qTo.y;
456 q.z += t1 * qTo.z;
457 q.w += t1 * qTo.w;
458
459 q.Normalize();
460
461 return q;
462}

476template <typename Type>
477void ezQuatTemplate<Type>::GetAsEulerAngles(ezAngle& out_x, ezAngle& out_y, ezAngle& out_z) const
478{
479 // Taken from https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles
480 // and http://www.euclideanspace.com/maths/geometry/rotations/conversions/quaternionToEuler/
481 // adapted to our convention (yaw->pitch->roll, ZYX order or 3-2-1 order)
482
483 auto& yaw = out_z;
484 auto& pitch = out_y;
485 auto& roll = out_x;
486
487 const double fSingularityTest = w * y - z * x;
488 const double fSingularityThreshold = 0.4999995;
489
490 if (fSingularityTest > fSingularityThreshold) // singularity at north pole
491 {
492 yaw = -2.0f * ezMath::ATan2(x, w);
493 pitch = ezAngle::MakeFromDegree(90.0f);
494 roll = ezAngle::MakeFromDegree(0.0f);
495 }
496 else if (fSingularityTest < -fSingularityThreshold) // singularity at south pole
497 {
498 yaw = 2.0f * ezMath::ATan2(x, w);
499 pitch = ezAngle::MakeFromDegree(-90.0f);
500 roll = ezAngle::MakeFromDegree(0.0f);
501 }
502 else
503 {
504 // yaw (z-axis rotation)
505 const double siny = 2.0 * (w * z + x * y);
506 const double cosy = 1.0 - 2.0 * (y * y + z * z);
507 yaw = ezMath::ATan2((float)siny, (float)cosy);
508
509 // pitch (y-axis rotation)
510 pitch = ezMath::ASin(2.0f * (float)fSingularityTest);
511
512 // roll (x-axis rotation)
513 const double sinr = 2.0 * (w * x + y * z);
514 const double cosr = 1.0 - 2.0 * (x * x + y * y);
515 roll = ezMath::ATan2((float)sinr, (float)cosr);
516 }
517}

519template <typename Type>
520ezQuatTemplate<Type> ezQuatTemplate<Type>::MakeFromEulerAngles(const ezAngle& x, const ezAngle& y, const ezAngle& z)
521{
524 const auto& yaw = z;
525 const auto& pitch = y;
526 const auto& roll = x;
527 const double cy = ezMath::Cos(yaw * 0.5);
528 const double sy = ezMath::Sin(yaw * 0.5);
529 const double cp = ezMath::Cos(pitch * 0.5);
530 const double sp = ezMath::Sin(pitch * 0.5);
531 const double cr = ezMath::Cos(roll * 0.5);
532 const double sr = ezMath::Sin(roll * 0.5);
533
534 ezQuatTemplate<Type> q;
535 q.w = (float)(cy * cp * cr + sy * sp * sr);
536 q.x = (float)(cy * cp * sr - sy * sp * cr);
537 q.y = (float)(cy * sp * cr + sy * cp * sr);
538 q.z = (float)(sy * cp * cr - cy * sp * sr);
539 return q;

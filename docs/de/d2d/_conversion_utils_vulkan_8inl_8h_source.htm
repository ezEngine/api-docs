351EZ_ALWAYS_INLINE vk::PipelineStageFlags ezConversionUtilsVulkan::GetPipelineStages(ezBitflags<ezGALShaderStageFlags> stages)
352{
353 vk::PipelineStageFlags res;
354 for (int i = 0; i < ezGALShaderStage::ENUM_COUNT; ++i)
355 {
356 ezGALShaderStageFlags::Enum flag = ezGALShaderStageFlags::MakeFromShaderStage((ezGALShaderStage::Enum)i);
357 if (stages.IsSet(flag))
358 {
359 res |= GetPipelineStage((ezGALShaderStage::Enum)i);
360 }
361 }
362 return res;
363}

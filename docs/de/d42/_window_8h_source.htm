57 struct ezWindowHandle
58 {
59  enum class Type
60  {
61  Invalid = 0,
62  GLFW = 1, // Used by the runtime
63  XCB = 2 // Used by the editor
64  };
65 
66  Type type;
67  union
68  {
69  GLFWwindow* glfwWindow;
70  ezXcbWindowHandle xcbWindow;
71  };
72 
73  bool operator==(ezWindowHandle& rhs)
74  {
75  if (type != rhs.type)
76  return false;
77 
78  if (type == Type::GLFW)
79  {
80  return glfwWindow == rhs.glfwWindow;
81  }
82  else
83  {
84  // We don't compare the connection because we only want to know if we reference the same window.
85  return xcbWindow.m_Window == rhs.xcbWindow.m_Window;
86  }
87  }
88 };

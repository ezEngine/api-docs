148template <typename IdType, typename ValueType>
149void ezIdTableBase<IdType, ValueType>::Reserve(IndexType capacity)
150{
151 if (m_Capacity >= capacity + CAPACITY_ALIGNMENT)
152 return;
153
154 const ezUInt64 uiCurCap64 = static_cast<ezUInt64>(this->m_Capacity);
155 ezUInt64 uiNewCapacity64 = uiCurCap64 + (uiCurCap64 / 2);
156
157 uiNewCapacity64 = ezMath::Max<ezUInt64>(uiNewCapacity64, capacity + CAPACITY_ALIGNMENT);
158
159 // the maximum value must leave room for the capacity alignment computation below (without overflowing the 32 bit range)
160 uiNewCapacity64 = ezMath::Min<ezUInt64>(uiNewCapacity64, 0xFFFFFFFFllu - (CAPACITY_ALIGNMENT - 1));
161
162 uiNewCapacity64 = (uiNewCapacity64 + (CAPACITY_ALIGNMENT - 1)) & ~(CAPACITY_ALIGNMENT - 1);
163
164 SetCapacity(static_cast<IndexType>(uiNewCapacity64 & 0xFFFFFFFF));
165}

242template <typename IdType, typename ValueType>
243bool ezIdTableBase<IdType, ValueType>::Remove(const IdType id, ValueType* out_pOldValue /*= nullptr*/)
244{
245 if (m_Capacity <= id.m_InstanceIndex)
246 return false;
247
248 const IndexType uiIndex = id.m_InstanceIndex;
249 Entry& entry = m_pEntries[uiIndex];
250 if (!entry.id.IsIndexAndGenerationEqual(id))
251 return false;
252
253 if (out_pOldValue != nullptr)
254 *out_pOldValue = std::move(m_pEntries[uiIndex].value);
255
256 ezMemoryUtils::Destruct(&entry.value, 1);
257
258 entry.id.m_InstanceIndex = m_pEntries[m_FreelistEnqueue].id.m_InstanceIndex;
259 ++entry.id.m_Generation;
260
261 // at wrap around, prevent generation from becoming 0, to ensure that a zero initialized array could ever contain a valid ID
262 if (entry.id.m_Generation == 0)
263 entry.id.m_Generation = 1;
264
265 m_pEntries[m_FreelistEnqueue].id.m_InstanceIndex = static_cast<decltype(entry.id.m_InstanceIndex)>(uiIndex);
266 m_FreelistEnqueue = uiIndex;
267
268 --m_Count;
269 return true;
270}

365template <typename IdType, typename ValueType>
366bool ezIdTableBase<IdType, ValueType>::IsFreelistValid() const
367{
368 if (m_pEntries == nullptr)
369 return true;
370
371 IndexType uiIndex = m_FreelistDequeue;
372 const Entry* pEntry = m_pEntries + uiIndex;
373
374 while (pEntry->id.m_InstanceIndex < m_Capacity)
375 {
376 uiIndex = pEntry->id.m_InstanceIndex;
377 pEntry = m_pEntries + uiIndex;
378 }
379
380 return uiIndex == m_FreelistEnqueue;
381}

385template <typename IdType, typename ValueType>
386void ezIdTableBase<IdType, ValueType>::SetCapacity(IndexType uiCapacity)
387{
388 Entry* pNewEntries = EZ_NEW_RAW_BUFFER(m_pAllocator, Entry, (size_t)uiCapacity);
389
390 for (IndexType i = 0; i < m_Capacity; ++i)
391 {
392 pNewEntries[i].id = m_pEntries[i].id;
393
394 if (m_pEntries[i].id.m_InstanceIndex == i)
395 {
396 ezMemoryUtils::RelocateConstruct(&pNewEntries[i].value, &m_pEntries[i].value, 1);
397 }
398 }
399
400 EZ_DELETE_RAW_BUFFER(m_pAllocator, m_pEntries);
401 m_pEntries = pNewEntries;
402
403 InitializeFreelist(m_Capacity, uiCapacity);
404 m_Capacity = uiCapacity;
405}

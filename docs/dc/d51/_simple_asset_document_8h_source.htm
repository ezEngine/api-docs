89 // Index based remapping ignores address identity and solely uses the object's parent index to define
90 // its guid. Set it to true if the native changes are complete clear and replace operations and
91 // not incremental changes to the existing data.
92 void ApplyNativePropertyChangesToObjectManager(bool bForceIndexBasedRemapping = false)
93 {
94 EZ_PROFILE_SCOPE("ApplyNativePropertyChangesToObjectManager");
95 // Create object manager graph
96 ezAbstractObjectGraph origGraph;
97 ezAbstractObjectNode* pOrigRootNode = nullptr;
98 {
99 ezDocumentObjectConverterWriter writer(&origGraph, this->GetObjectManager());
100 pOrigRootNode = writer.AddObjectToGraph(GetPropertyObject());
101 }
102
103 // Create native object graph
104 ezAbstractObjectGraph graph;
105 ezAbstractObjectNode* pRootNode = nullptr;
106 {
107 // The ezApplyNativePropertyChangesContext takes care of generating guids for native pointers that match those
108 // of the object manager.
109 ezApplyNativePropertyChangesContext nativeChangesContext(m_Context, origGraph);
110 ezRttiConverterWriter rttiConverter(&graph, &nativeChangesContext, true, true);
111 nativeChangesContext.RegisterObject(pOrigRootNode->GetGuid(), ezGetStaticRTTI<PropertyType>(), GetProperties());
112 pRootNode = rttiConverter.AddObjectToGraph(GetProperties(), "Object");
113 }
114
115 // Remapping is no longer necessary as ezApplyNativePropertyChangesContext takes care of mapping to the original nodes.
116 // However, if the native changes are done like clear+rebuild everything, then no original object will be found and
117 // every pointer will be deleted and re-created. Forcing the remapping (which works entirely via index and ignores
118 // pointer addresses) will yield better results (e.g. no changes on two back-to -back transform calls).
119 if (bForceIndexBasedRemapping)
120 {
121 // Remap native guids so they match the object manager (stuff like embedded classes will not have a guid on the native side).
122 graph.ReMapNodeGuidsToMatchGraph(pRootNode, origGraph, pOrigRootNode);
123 }
124
125 ezDeque<ezAbstractGraphDiffOperation> diffResult;
126 graph.CreateDiffWithBaseGraph(origGraph, diffResult);
127
128 // if index-based remapping is used, we MUST send a change event of some kind
129 // since the underlying data structures (memory locations) might have been changed,
130 // even if there is no actual change to the content
131 // the command history will detect that there was no change and actually send a "TransactionCanceled" event, but that is enough for other code to react to
132 if (!diffResult.IsEmpty() || bForceIndexBasedRemapping)
133 {
134 // As we messed up the native side the object mirror is no longer synced and needs to be destroyed.
135 m_ObjectMirror.Clear();
136
137 // Apply diff while object mirror is down.
138 this->GetObjectAccessor()->StartTransaction("Apply Native Property Changes to Object");
139
140 ezDocumentObjectConverterReader::ApplyDiffToObject(this->GetObjectAccessor(), GetPropertyObject(), diffResult);
141
142 // Re-apply document
143 m_ObjectMirror.SendDocument();
144
145 this->GetObjectAccessor()->FinishTransaction();
146 }
147 }

22  virtual ezUuid GenerateObjectGuid(const ezUuid& parentGuid, const ezAbstractProperty* pProp, ezVariant index, void* pObject) const override
23  {
24  ezUuid guid;
25  if (pProp->GetFlags().IsSet(ezPropertyFlags::PointerOwner))
26  {
27  // If the object is already known by the native context (a pointer that existed before the native changes)
28  // we can just return it. Any other pointer will get a new guid assigned.
29  guid = m_nativeContext.GetObjectGUID(pProp->GetSpecificType(), pObject);
30  if (guid.IsValid())
31  return guid;
32  }
33  else if (pProp->GetFlags().IsSet(ezPropertyFlags::Class))
34  {
35  // In case of by-value classes we lookup the guid in the object manager graph by using
36  // the index as the identify of the object. If the index is not valid (e.g. the array was expanded by native changes)
37  // a new guid is assigned.
38  if (const ezAbstractObjectNode* originalNode = m_originalGraph.GetNode(parentGuid))
39  {
40  if (const ezAbstractObjectNode::Property* originalProp = originalNode->FindProperty(pProp->GetPropertyName()))
41  {
42  switch (pProp->GetCategory())
43  {
44  case ezPropertyCategory::Member:
45  {
46  if (originalProp->m_Value.IsA<ezUuid>() && originalProp->m_Value.Get<ezUuid>().IsValid())
47  return originalProp->m_Value.Get<ezUuid>();
48  }
49  break;
50  case ezPropertyCategory::Array:
51  {
52  ezUInt32 uiIndex = index.Get<ezUInt32>();
53  if (originalProp->m_Value.IsA<ezVariantArray>())
54  {
55  const ezVariantArray& values = originalProp->m_Value.Get<ezVariantArray>();
56  if (uiIndex < values.GetCount())
57  {
58  const auto& originalElemValue = values[uiIndex];
59  if (originalElemValue.IsA<ezUuid>() && originalElemValue.Get<ezUuid>().IsValid())
60  return originalElemValue.Get<ezUuid>();
61  }
62  }
63  }
64  break;
65  case ezPropertyCategory::Map:
66  {
67  const ezString& sIndex = index.Get<ezString>();
68  if (originalProp->m_Value.IsA<ezVariantDictionary>())
69  {
70  const ezVariantDictionary& values = originalProp->m_Value.Get<ezVariantDictionary>();
71  if (values.Contains(sIndex))
72  {
73  const auto& originalElemValue = *values.GetValue(sIndex);
74  if (originalElemValue.IsA<ezUuid>() && originalElemValue.Get<ezUuid>().IsValid())
75  return originalElemValue.Get<ezUuid>();
76  }
77  }
78  }
79  break;
80 
81  default:
82  break;
83  }
84  }
85  }
86  }
87  guid.CreateNewUuid();
88  return guid;
89  }
90 
91 private:
92  ezRttiConverterContext& m_nativeContext;
93  const ezAbstractObjectGraph& m_originalGraph;
94 };

164  // Index based remapping ignores address identity and solely uses the object's parent index to define
165  // its guid. Set it to true if the native changes are complete clear and replace operations and
166  // not incremental changes to the existing data.
167  void ApplyNativePropertyChangesToObjectManager(bool bForceIndexBasedRemapping = false)
168  {
169  EZ_PROFILE_SCOPE("ApplyNativePropertyChangesToObjectManager");
170  // Create object manager graph
171  ezAbstractObjectGraph origGraph;
172  ezAbstractObjectNode* pOrigRootNode = nullptr;
173  {
174  ezDocumentObjectConverterWriter writer(&origGraph, this->GetObjectManager());
175  pOrigRootNode = writer.AddObjectToGraph(GetPropertyObject());
176  }
177 
178  // Create native object graph
179  ezAbstractObjectGraph graph;
180  ezAbstractObjectNode* pRootNode = nullptr;
181  {
182  // The ezApplyNativePropertyChangesContext takes care of generating guids for native pointers that match those
183  // of the object manager.
184  ezApplyNativePropertyChangesContext nativeChangesContext(m_Context, origGraph);
185  ezRttiConverterWriter rttiConverter(&graph, &nativeChangesContext, true, true);
186  nativeChangesContext.RegisterObject(pOrigRootNode->GetGuid(), ezGetStaticRTTI<PropertyType>(), GetProperties());
187  pRootNode = rttiConverter.AddObjectToGraph(GetProperties(), "Object");
188  }
189 
190  // Remapping is no longer necessary as ezApplyNativePropertyChangesContext takes care of mapping to the original nodes.
191  // However, if the native changes are done like clear+rebuild everything, then no original object will be found and
192  // every pointer will be deleted and re-created. Forcing the remapping (which works entirely via index and ignores
193  // pointer addresses) will yield better results (e.g. no changes on two back-to -back transform calls).
194  if (bForceIndexBasedRemapping)
195  {
196  // Remap native guids so they match the object manager (stuff like embedded classes will not have a guid on the native side).
197  graph.ReMapNodeGuidsToMatchGraph(pRootNode, origGraph, pOrigRootNode);
198  }
199 
200  ezDeque<ezAbstractGraphDiffOperation> diffResult;
201  graph.CreateDiffWithBaseGraph(origGraph, diffResult);
202 
203  if (!diffResult.IsEmpty())
204  {
205  // As we messed up the native side the object mirror is no longer synced and needs to be destroyed.
206  m_ObjectMirror.Clear();
207 
208  // Apply diff while object mirror is down.
209  this->GetObjectAccessor()->StartTransaction("Apply Native Property Changes to Object");
210 
211  ezDocumentObjectConverterReader::ApplyDiffToObject(this->GetObjectAccessor(), GetPropertyObject(), diffResult);
212 
213  // Re-apply document
214  m_ObjectMirror.SendDocument();
215 
216  this->GetObjectAccessor()->FinishTransaction();
217  }
218  }

66template <typename CompatibleKeyType>
67ezUInt32 ezArrayMapBase<KEY, VALUE>::Find(const CompatibleKeyType& key) const
68{
69 if (!m_bSorted)
70 {
71 m_Data.Sort();
72 }
73
74 ezUInt32 lb = 0;
75 ezUInt32 ub = m_Data.GetCount();
76
77 while (lb < ub)
78 {
79 const ezUInt32 middle = lb + ((ub - lb) >> 1);
80
81 if (m_Data[middle].key < key)
82 {
83 lb = middle + 1;
84 }
85 else if (key < m_Data[middle].key)
86 {
87 ub = middle;
88 }
89 else // equal
90 {
91 return middle;
92 }
93 }
94
95 return ezInvalidIndex;
96}

260{
261 ezUInt32 atpos = LowerBound(key);
262
263 if (atpos == ezInvalidIndex)
264 return false;

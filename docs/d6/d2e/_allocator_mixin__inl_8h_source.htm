62template <typename A, ezAllocatorTrackingMode TrackingMode>
63void* ezInternal::ezAllocatorImpl<A, TrackingMode>::Allocate(size_t uiSize, size_t uiAlign, ezMemoryUtils::DestructorFunction destructorFunc)
64{
65 EZ_IGNORE_UNUSED(destructorFunc);
66
67 // zero size allocations always return nullptr without tracking (since deallocate nullptr is ignored)
68 if (uiSize == 0)
69 return nullptr;
70
71 EZ_ASSERT_DEBUG(ezMath::IsPowerOf2((ezUInt32)uiAlign), "Alignment must be power of two");
72
73 [[maybe_unused]] ezTime fAllocationTime;
74
75 if constexpr (TrackingMode >= ezAllocatorTrackingMode::AllocationStats)
76 {
77 fAllocationTime = ezTime::Now();
78 }
79
80 void* ptr = m_allocator.Allocate(uiSize, uiAlign);
81 EZ_ASSERT_DEV(ptr != nullptr, "Could not allocate {0} bytes. Out of memory?", uiSize);
82
83 if constexpr (TrackingMode >= ezAllocatorTrackingMode::AllocationStats)
84 {
85 ezMemoryTracker::AddAllocation(this->m_Id, TrackingMode, ptr, uiSize, uiAlign, ezTime::Now() - fAllocationTime);
86 }
87
88 return ptr;
89}

152template <typename A, ezAllocatorTrackingMode TrackingMode>
153void* ezInternal::ezAllocatorMixinReallocate<A, TrackingMode, true>::Reallocate(void* pPtr, size_t uiCurrentSize, size_t uiNewSize, size_t uiAlign)
154{
155 [[maybe_unused]] ezTime fAllocationTime;
156
157 if constexpr (TrackingMode >= ezAllocatorTrackingMode::AllocationStats)
158 {
159 ezMemoryTracker::RemoveAllocation(this->m_Id, pPtr);
160 fAllocationTime = ezTime::Now();
161 }
162
163 void* pNewMem = this->m_allocator.Reallocate(pPtr, uiCurrentSize, uiNewSize, uiAlign);
164
165 if constexpr (TrackingMode >= ezAllocatorTrackingMode::AllocationStats)
166 {
167 ezMemoryTracker::AddAllocation(this->m_Id, TrackingMode, pNewMem, uiNewSize, uiAlign, ezTime::Now() - fAllocationTime);
168 }
169
170 return pNewMem;
171}

327 template <typename Type>
328 Type ezBoundingBoxTemplate<Type>::GetDistanceSquaredTo(const ezBoundingBoxTemplate<Type>& rhs) const
329 {
330  // This will return zero for overlapping boxes
331 
332  EZ_NAN_ASSERT(this);
333  EZ_NAN_ASSERT(&rhs);
334 
335  Type fDistSQR = 0.0f;
336 
337  {
338  if (rhs.m_vMin.x > m_vMax.x)
339  {
340  fDistSQR += ezMath::Square(rhs.m_vMin.x - m_vMax.x);
341  }
342  else if (rhs.m_vMax.x < m_vMin.x)
343  {
344  fDistSQR += ezMath::Square(m_vMin.x - rhs.m_vMax.x);
345  }
346  }
347 
348  {
349  if (rhs.m_vMin.y > m_vMax.y)
350  {
351  fDistSQR += ezMath::Square(rhs.m_vMin.y - m_vMax.y);
352  }
353  else if (rhs.m_vMax.y < m_vMin.y)
354  {
355  fDistSQR += ezMath::Square(m_vMin.y - rhs.m_vMax.y);
356  }
357  }
358 
359  {
360  if (rhs.m_vMin.z > m_vMax.z)
361  {
362  fDistSQR += ezMath::Square(rhs.m_vMin.z - m_vMax.z);
363  }
364  else if (rhs.m_vMax.z < m_vMin.z)
365  {
366  fDistSQR += ezMath::Square(m_vMin.z - rhs.m_vMax.z);
367  }
368  }
369 
370  return fDistSQR;
371 }

379 template <typename Type>
380 bool ezBoundingBoxTemplate<Type>::GetRayIntersection(
381  const ezVec3Template<Type>& vStartPos, const ezVec3Template<Type>& vRayDir, Type* out_fIntersection, ezVec3Template<Type>* out_vIntersection) const
382 {
383  // This code was taken from: http://people.csail.mit.edu/amy/papers/box-jgt.pdf
384  // "An Efficient and Robust Ray-Box Intersection Algorithm"
385  // Contrary to previous implementation, this one actually works with ray/box configurations
386  // that produce division by zero and multiplication with infinity (which can produce NaNs).
387 
388  EZ_ASSERT_DEBUG(ezMath::SupportsInfinity<Type>(), "This type does not support infinite values, which is required for this algorithm.");
389  EZ_ASSERT_DEBUG(vStartPos.IsValid(), "Ray start position must be valid.");
390  EZ_ASSERT_DEBUG(vRayDir.IsValid(), "Ray direction must be valid.");
391 
392  EZ_NAN_ASSERT(this);
393 
394  float tMin, tMax;
395 
396  // Compare along X and Z axis, find intersection point
397  {
398  float tMinY, tMaxY;
399 
400  const float fDivX = 1.0f / vRayDir.x;
401  const float fDivY = 1.0f / vRayDir.y;
402 
403  if (vRayDir.x >= 0.0f)
404  {
405  tMin = (m_vMin.x - vStartPos.x) * fDivX;
406  tMax = (m_vMax.x - vStartPos.x) * fDivX;
407  }
408  else
409  {
410  tMin = (m_vMax.x - vStartPos.x) * fDivX;
411  tMax = (m_vMin.x - vStartPos.x) * fDivX;
412  }
413 
414  if (vRayDir.y >= 0.0f)
415  {
416  tMinY = (m_vMin.y - vStartPos.y) * fDivY;
417  tMaxY = (m_vMax.y - vStartPos.y) * fDivY;
418  }
419  else
420  {
421  tMinY = (m_vMax.y - vStartPos.y) * fDivY;
422  tMaxY = (m_vMin.y - vStartPos.y) * fDivY;
423  }
424 
425  if (tMin > tMaxY || tMinY > tMax)
426  return false;
427 
428  if (tMinY > tMin)
429  tMin = tMinY;
430  if (tMaxY < tMax)
431  tMax = tMaxY;
432  }
433 
434  // Compare along Z axis and previous result, find intersection point
435  {
436  float tMinZ, tMaxZ;
437 
438  const float fDivZ = 1.0f / vRayDir.z;
439 
440  if (vRayDir.z >= 0.0f)
441  {
442  tMinZ = (m_vMin.z - vStartPos.z) * fDivZ;
443  tMaxZ = (m_vMax.z - vStartPos.z) * fDivZ;
444  }
445  else
446  {
447  tMinZ = (m_vMax.z - vStartPos.z) * fDivZ;
448  tMaxZ = (m_vMin.z - vStartPos.z) * fDivZ;
449  }
450 
451  if (tMin > tMaxZ || tMinZ > tMax)
452  return false;
453 
454  if (tMinZ > tMin)
455  tMin = tMinZ;
456  if (tMaxZ < tMax)
457  tMax = tMaxZ;
458  }
459 
460  // rays that start inside the box are considered as not hitting the box
461  if (tMax <= 0.0f)
462  return false;
463 
464  if (out_fIntersection)
465  *out_fIntersection = tMin;
466 
467  if (out_vIntersection)
468  *out_vIntersection = vStartPos + tMin * vRayDir;
469 
470  return true;
471 }

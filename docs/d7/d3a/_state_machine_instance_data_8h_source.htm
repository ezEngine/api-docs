119  m_InstanceDataOffsets.PushBack(uiOffset);
120  }

126 
127  ezUInt32 uiBaseOffset = ezMemoryUtils::AlignSize(out_desc.m_uiTypeSize, uiMaxAlignment);
128  m_InstanceDataOffsets.GetUserData<ezUInt32>() = uiBaseOffset;
129 
130  out_desc.m_uiTypeSize = uiBaseOffset + m_InstanceDataAllocator.GetTotalDataSize();
131  out_desc.m_uiTypeAlignment = ezMath::Max(out_desc.m_uiTypeAlignment, uiMaxAlignment);

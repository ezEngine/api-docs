103template <typename Class, typename Type>
104class ezAccessorProperty : public ezTypedMemberProperty<typename ezTypeTraits<Type>::NonConstReferenceType>
105{
106public:
107 using RealType = typename ezTypeTraits<Type>::NonConstReferenceType;
108 using GetterFunc = Type (Class::*)() const;
109 using SetterFunc = void (Class::*)(Type value);
110
112 ezAccessorProperty(const char* szPropertyName, GetterFunc getter, SetterFunc setter)
113 : ezTypedMemberProperty<RealType>(szPropertyName)
114 {
115 EZ_ASSERT_DEBUG(getter != nullptr, "The getter of a property cannot be nullptr.");
116
117 m_Getter = getter;
118 m_Setter = setter;
119
120 if (m_Setter == nullptr)
121 ezAbstractMemberProperty::m_Flags.Add(ezPropertyFlags::ReadOnly);
122 }
123
126 virtual void* GetPropertyPointer(const void* pInstance) const override
127 {
128 EZ_IGNORE_UNUSED(pInstance);
129
130 // No access to sub-properties, if we have accessors for this property
131 return nullptr;
132 }
133
135 virtual RealType GetValue(const void* pInstance) const override // [tested]
136 {
137 return (static_cast<const Class*>(pInstance)->*m_Getter)();
138 }
139
143 virtual void SetValue(void* pInstance, RealType value) const override // [tested]
144 {
145 EZ_ASSERT_DEV(m_Setter != nullptr, "The property '{0}' has no setter function, thus it is read-only.", ezAbstractProperty::GetPropertyName());
146
147 if (m_Setter)
148 (static_cast<Class*>(pInstance)->*m_Setter)(value);
149 }
150
151private:
152 GetterFunc m_Getter;
153 SetterFunc m_Setter;
154};

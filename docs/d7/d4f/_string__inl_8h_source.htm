89 template <ezUInt16 Size>
90 void ezHybridStringBase<Size>::operator=(const char* szString)
91 {
92  ezUInt32 uiElementCount = 0;
93  ezStringUtils::GetCharacterAndElementCount(szString, m_uiCharacterCount, uiElementCount);
94 
95  if (szString + uiElementCount < m_Data.GetData() || szString >= m_Data.GetData() + m_Data.GetCount())
96  {
97  // source string is outside our own memory, so no overlapped copy
98  }
99  else
100  {
101  // source string overlaps with our own memory -> we can't increase the size of our memory, as that might invalidate the source data
102  EZ_ASSERT_DEBUG(uiElementCount < m_Data.GetCount(), "Invalid copy of overlapping string data.");
103  }
104 
105  m_Data.SetCountUninitialized(uiElementCount + 1);
106  ezStringUtils::Copy(&m_Data[0], uiElementCount + 1, szString);
107 }

148 ezStringView ezHybridStringBase<Size>::GetSubString(ezUInt32 uiFirstCharacter, ezUInt32 uiNumCharacters) const
149 {
150  EZ_ASSERT_DEV(uiFirstCharacter + uiNumCharacters <= m_uiCharacterCount,
151  "The string only has {0} characters, cannot get a sub-string up to character {1}.", m_uiCharacterCount, uiFirstCharacter + uiNumCharacters);
152 
153  const char* szStart = GetData();
154  ezUnicodeUtils::MoveToNextUtf8(szStart, uiFirstCharacter);
155 
156  const char* szEnd = szStart;
157  ezUnicodeUtils::MoveToNextUtf8(szEnd, uiNumCharacters);
158 
159  return ezStringView(szStart, szEnd);

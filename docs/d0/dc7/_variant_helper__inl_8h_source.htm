179  static void To(const ezVariant& value, bool& result, bool& bSuccessful)
180  {
181  bSuccessful = true;
182 
183  if (value.GetType() <= ezVariant::Type::Double)
184  result = value.ConvertNumber<ezInt32>() != 0;
185  else if (value.GetType() == ezVariant::Type::String || value.GetType() == ezVariant::Type::HashedString)
186  {
187  ezStringView s = value.IsA<ezString>() ? value.Cast<ezString>().GetView() : value.Cast<ezHashedString>().GetView();
188  if (ezConversionUtils::StringToBool(s, result) == EZ_FAILURE)
189  {
190  result = false;
191  bSuccessful = false;
192  }
193  }
194  else
195  EZ_REPORT_FAILURE("Conversion to bool failed");
196  }

234 template <typename BlockStorageAllocator>
235 void ezTagSetTemplate<BlockStorageAllocator>::Set(const ezTag& tag)
236 {
237  EZ_ASSERT_DEV(tag.IsValid(), "Only valid tags can be set in a tag set!");
238 
239  if (m_TagBlocks.IsEmpty())
240  {
241  Reallocate(tag.m_uiBlockIndex, tag.m_uiBlockIndex);
242  }
243  else if (IsTagInAllocatedRange(tag) == false)
244  {
245  const ezUInt32 uiNewBlockStart = ezMath::Min<ezUInt32>(tag.m_uiBlockIndex, GetTagBlockStart());
246  const ezUInt32 uiNewBlockEnd = ezMath::Max<ezUInt32>(tag.m_uiBlockIndex, GetTagBlockEnd());
247 
248  Reallocate(uiNewBlockStart, uiNewBlockEnd);
249  }
250 
251  ezUInt64& tagBlock = m_TagBlocks[tag.m_uiBlockIndex - GetTagBlockStart()];
252 
253  const ezUInt64 bitMask = EZ_BIT(tag.m_uiBitIndex);
254  const bool bBitWasSet = ((tagBlock & bitMask) != 0);
255 
256  tagBlock |= bitMask;
257 
258  if (!bBitWasSet)
259  {
260  IncreaseTagCount();
261  }
262 }

300 template <typename BlockStorageAllocator>
301 bool ezTagSetTemplate<BlockStorageAllocator>::IsAnySet(const ezTagSetTemplate& otherSet) const
302 {
303  // If any of the sets is empty nothing can match
304  if (IsEmpty() || otherSet.IsEmpty())
305  return false;
306 
307  // Calculate range to compare
308  const ezUInt32 uiMaxBlockStart = ezMath::Max(GetTagBlockStart(), otherSet.GetTagBlockStart());
309  const ezUInt32 uiMinBlockEnd = ezMath::Min(GetTagBlockEnd(), otherSet.GetTagBlockEnd());
310 
311  if (uiMaxBlockStart > uiMinBlockEnd)
312  return false;
313 
314  for (ezUInt32 i = uiMaxBlockStart; i < uiMinBlockEnd; ++i)
315  {
316  const ezUInt32 uiThisBlockStorageIndex = i - GetTagBlockStart();
317  const ezUInt32 uiOtherBlockStorageIndex = i - otherSet.GetTagBlockStart();
318 
319  if ((m_TagBlocks[uiThisBlockStorageIndex] & otherSet.m_TagBlocks[uiOtherBlockStorageIndex]) != 0)
320  {
321  return true;
322  }
323  }
324 
325  return false;
326 }

381 template <typename BlockStorageAllocator>
382 void ezTagSetTemplate<BlockStorageAllocator>::Reallocate(ezUInt32 uiNewTagBlockStart, ezUInt32 uiNewMaxBlockIndex)
383 {
384  EZ_ASSERT_DEV(uiNewTagBlockStart < ezSmallInvalidIndex, "Tag block start is too big");
385  const ezUInt16 uiNewBlockArraySize = static_cast<ezUInt16>((uiNewMaxBlockIndex - uiNewTagBlockStart) + 1);
386 
387  // Early out for non-filled tag sets
388  if (m_TagBlocks.IsEmpty())
389  {
390  m_TagBlocks.SetCount(uiNewBlockArraySize);
391  SetTagBlockStart(static_cast<ezUInt16>(uiNewTagBlockStart));
392 
393  return;
394  }
395 
396  EZ_ASSERT_DEBUG(uiNewTagBlockStart <= GetTagBlockStart(), "New block start must be smaller or equal to current block start!");
397 
398  ezSmallArray<ezUInt64, 32, BlockStorageAllocator> helperArray;
399  helperArray.SetCount(uiNewBlockArraySize);
400 
401  const ezUInt32 uiOldBlockStartOffset = GetTagBlockStart() - uiNewTagBlockStart;
402 
403  // Copy old data to the new array
404  ezMemoryUtils::Copy(helperArray.GetData() + uiOldBlockStartOffset, m_TagBlocks.GetData(), m_TagBlocks.GetCount());
405 
406  // Use array ptr copy assignment so it doesn't modify the user data in m_TagBlocks
407  m_TagBlocks = helperArray.GetArrayPtr();
408  SetTagBlockStart(static_cast<ezUInt16>(uiNewTagBlockStart));
409 }

39template <typename ArrayType, typename ValueType>
40ezResult ezDeduplicationWriteContext::WriteArray(ezStreamWriter& inout_stream, const ezArrayBase<ValueType, ArrayType>& array)
41{
42 const ezUInt64 uiCount = array.GetCount();
43 EZ_SUCCEED_OR_RETURN(inout_stream.WriteQWordValue(&uiCount));
44
45 for (ezUInt32 i = 0; i < static_cast<ezUInt32>(uiCount); ++i)

47 EZ_SUCCEED_OR_RETURN(WriteObject(inout_stream, array[i]));

67template <typename KeyType, typename ValueType, typename Comparer>
68ezResult ezDeduplicationWriteContext::WriteMap(ezStreamWriter& inout_stream, const ezMapBase<KeyType, ValueType, Comparer>& map, WriteMapMode mode)
69{
70 const ezUInt64 uiWriteSize = map.GetCount();
71 EZ_SUCCEED_OR_RETURN(inout_stream.WriteQWordValue(&uiWriteSize));
72
73 if (mode == WriteMapMode::DedupKey)
74 {
75 for (auto It = map.GetIterator(); It.IsValid(); ++It)
76 {
77 EZ_SUCCEED_OR_RETURN(WriteObject(inout_stream, It.Key()));
78 EZ_SUCCEED_OR_RETURN(ezStreamWriterUtil::Serialize<ValueType>(inout_stream, It.Value()));
79 }
80 }
81 else if (mode == WriteMapMode::DedupValue)
82 {
83 for (auto It = map.GetIterator(); It.IsValid(); ++It)
84 {
85 EZ_SUCCEED_OR_RETURN(ezStreamWriterUtil::Serialize<KeyType>(inout_stream, It.Key()));
86 EZ_SUCCEED_OR_RETURN(WriteObject(inout_stream, It.Value()));
87 }
88 }
89 else
90 {
91 for (auto It = map.GetIterator(); It.IsValid(); ++It)
92 {
93 EZ_SUCCEED_OR_RETURN(WriteObject(inout_stream, It.Key()));
94 EZ_SUCCEED_OR_RETURN(WriteObject(inout_stream, It.Value()));
95 }
96 }
97
98 return EZ_SUCCESS;
99}

101template <typename T>
102ezResult ezDeduplicationWriteContext::WriteObjectInternal(ezStreamWriter& stream, const T* pObject)
103{
104 ezUInt32 uiIndex = ezInvalidIndex;
105
106 if (pObject)
107 {
108 bool bIsRealObject = !m_Objects.TryGetValue(pObject, uiIndex);
109 stream << bIsRealObject;
110
111 if (bIsRealObject)
112 {
113 uiIndex = m_Objects.GetCount();
114 m_Objects.Insert(pObject, uiIndex);
115
116 return ezStreamWriterUtil::Serialize<T>(stream, *pObject);
117 }
118 else
119 {
120 stream << uiIndex;
121 }
122 }
123 else
124 {
125 stream << false;
126 stream << ezInvalidIndex;
127 }
128
129 return EZ_SUCCESS;
130}

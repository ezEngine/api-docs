147template <typename Type>
148void ezBoundingBoxSphereTemplate<Type>::ExpandToInclude(const ezBoundingBoxSphereTemplate& rhs)
149{
150 ezBoundingBoxTemplate<Type> box;
151 box.m_vMin = m_vCenter - m_vBoxHalfExtents;
152 box.m_vMax = m_vCenter + m_vBoxHalfExtents;
153 box.ExpandToInclude(rhs.GetBox());
154
155 ezBoundingBoxSphereTemplate<Type> result = ezBoundingBoxSphereTemplate<Type>::MakeFromBox(box);
156
157 const float fSphereRadiusA = (m_vCenter - result.m_vCenter).GetLength() + m_fSphereRadius;
158 const float fSphereRadiusB = (rhs.m_vCenter - result.m_vCenter).GetLength() + rhs.m_fSphereRadius;
159
160 m_vCenter = result.m_vCenter;
161 m_fSphereRadius = ezMath::Min(result.m_fSphereRadius, ezMath::Max(fSphereRadiusA, fSphereRadiusB));
162 m_vBoxHalfExtents = result.m_vBoxHalfExtents;
163}

165template <typename Type>
166void ezBoundingBoxSphereTemplate<Type>::Transform(const ezMat4Template<Type>& mTransform)
167{
168 m_vCenter = mTransform.TransformPosition(m_vCenter);
169 const ezVec3Template<Type> Scale = mTransform.GetScalingFactors();
170 m_fSphereRadius *= ezMath::Max(Scale.x, Scale.y, Scale.z);
171
172 ezMat3Template<Type> mAbsRotation = mTransform.GetRotationalPart();
173 for (ezUInt32 i = 0; i < 9; ++i)
174 {
175 mAbsRotation.m_fElementsCM[i] = ezMath::Abs(mAbsRotation.m_fElementsCM[i]);
176 }
177
178 m_vBoxHalfExtents = mAbsRotation.TransformDirection(m_vBoxHalfExtents).CompMin(ezVec3(m_fSphereRadius));
179}

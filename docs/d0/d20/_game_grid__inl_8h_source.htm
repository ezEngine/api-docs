129template <class CellData>
130bool ezGameGrid<CellData>::GetRayIntersection(const ezVec3& vRayStartWorldSpace, const ezVec3& vRayDirNormalizedWorldSpace, float fMaxLength,
131 float& out_fIntersection, ezVec2I32& out_vCellCoord) const
132{
133 const ezVec3 vRayStart = m_mRotateToGridspace * (vRayStartWorldSpace - m_vWorldSpaceOrigin);
134 const ezVec3 vRayDir = m_mRotateToGridspace * vRayDirNormalizedWorldSpace;
135
136 ezVec3 vGridBox(m_uiGridSizeX, m_uiGridSizeY, 1.0f);
137
138 const ezBoundingBox localBox(ezVec3(0.0f), m_vLocalSpaceCellSize.CompMul(vGridBox));
139
140 if (localBox.Contains(vRayStart))
141 {
142 // if the ray is already inside the box, we know that a cell is hit
143 out_fIntersection = 0.0f;
144 }
145 else
146 {
147 if (!localBox.GetRayIntersection(vRayStart, vRayDir, &out_fIntersection, nullptr))
148 return false;
149
150 if (out_fIntersection > fMaxLength)
151 return false;
152 }
153
154 const ezVec3 vEnterPos = vRayStart + vRayDir * out_fIntersection;
155
156 const ezVec3 vCell = vEnterPos.CompMul(m_vInverseLocalSpaceCellSize);
157
158 // Without the Floor, the border case when the position is outside (-1 / -1) is not immediately detected
159 out_vCellCoord = ezVec2I32((ezInt32)ezMath::Floor(vCell.x), (ezInt32)ezMath::Floor(vCell.y));
160 out_vCellCoord.x = ezMath::Clamp(out_vCellCoord.x, 0, m_uiGridSizeX - 1);
161 out_vCellCoord.y = ezMath::Clamp(out_vCellCoord.y, 0, m_uiGridSizeY - 1);
162
163 return true;
164}

274template <class CellData, class EdgeData>
275ezUInt32 ezGameGridWithEdges<CellData, EdgeData>::ConvertCellCoordinateToEdgeIndex(const ezVec2I32& vCoord, ezGameGridCellEdge edge) const
276{
277 const ezUInt32 uiOffsetY = (this->m_uiGridSizeX + 1) * this->m_uiGridSizeY;
278
279 switch (edge)
280 {
281 case ezGameGridCellEdge::NegX:
282 return vCoord.y * (this->m_uiGridSizeX + 1) + vCoord.x;
283
284 case ezGameGridCellEdge::PosX:
285 return vCoord.y * (this->m_uiGridSizeX + 1) + vCoord.x + 1;
286
287 case ezGameGridCellEdge::NegY:
288 return uiOffsetY + vCoord.x * (this->m_uiGridSizeY + 1) + vCoord.y;
289
290 case ezGameGridCellEdge::PosY:
291 return uiOffsetY + vCoord.x * (this->m_uiGridSizeY + 1) + vCoord.y + 1;
292
293 EZ_DEFAULT_CASE_NOT_IMPLEMENTED;
294 }
295
296 return 0;

381template <class CellData, class EdgeData>
382bool ezGameGridWithEdges<CellData, EdgeData>::GetRayIntersectionWithEdge(const ezVec3& vRayStartWorldSpace,
383 const ezVec3& vRayDirNormalizedWorldSpace, float fMaxLength, float& out_fIntersection, ezVec2I32& out_vCellCoord, ezGameGridCellEdge& out_edge) const
384{
385 // First find which cell is hit
386 if (!this->GetRayIntersection(vRayStartWorldSpace, vRayDirNormalizedWorldSpace, fMaxLength, out_fIntersection, out_vCellCoord))
387 return false;
388
389 // Calculate the intersection point in world space
390 const ezVec3 vIntersectionWorld = vRayStartWorldSpace + vRayDirNormalizedWorldSpace * out_fIntersection;
391
392 // Transform intersection point to local space
393 const ezVec3 vIntersectionLocal = this->m_mRotateToGridspace * (vIntersectionWorld - this->m_vWorldSpaceOrigin);
394
395 // Get the cell center in local space
396 const ezVec3 vCellCenter = this->GetCellLocalSpaceCenter(out_vCellCoord);
397
398 // Calculate the offset from cell center to intersection point
399 const ezVec3 vOffset = vIntersectionLocal - vCellCenter;
400
401 // Determine which edge is closest based on the offset direction
402 // Compare absolute values to find the dominant axis
403 const float fAbsX = ezMath::Abs(vOffset.x);
404 const float fAbsY = ezMath::Abs(vOffset.y);
405
406 if (fAbsX > fAbsY)
407 {
408 // Closer to a vertical edge (NegX or PosX)
409 out_edge = (vOffset.x < 0.0f) ? ezGameGridCellEdge::NegX : ezGameGridCellEdge::PosX;
410 }
411 else
412 {
413 // Closer to a horizontal edge (NegY or PosY)
414 out_edge = (vOffset.y < 0.0f) ? ezGameGridCellEdge::NegY : ezGameGridCellEdge::PosY;
415 }
416
417 return true;
418}

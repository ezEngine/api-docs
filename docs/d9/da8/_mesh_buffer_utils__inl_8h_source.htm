33EZ_ALWAYS_INLINE ezResult ezMeshBufferUtils::EncodeColor(const ezColor& color, ezByteArrayPtr dest, ezGALResourceFormat::Enum destFormat, ezMeshVertexColorConversion::Enum conversion)
34{
35 ezVec4 finalColor = color.GetAsVec4();
36 if (conversion == ezMeshVertexColorConversion::LinearToSrgb)
37 {
38 finalColor = ezColor::LinearToGamma(finalColor.GetAsVec3()).GetAsVec4(finalColor.w);
39 }
40 else if (conversion == ezMeshVertexColorConversion::SrgbToLinear)
41 {
42 finalColor = ezColor::GammaToLinear(finalColor.GetAsVec3()).GetAsVec4(finalColor.w);
43 }
44
45 return EncodeFromVec4(finalColor, dest, destFormat);
46}

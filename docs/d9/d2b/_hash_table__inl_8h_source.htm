67 template <typename K, typename V, typename H>
68 void ezHashTableBase<K, V, H>::ConstIterator::Next()
69 {
70  // if we already iterated over the amount of valid elements that the hash-table stores, early out
71  if (m_uiCurrentCount >= m_hashTable->m_uiCount)
72  return;
73 
74  // increase the counter of how many elements we have seen
75  ++m_uiCurrentCount;
76  // increase the index of the element to look at
77  ++m_uiCurrentIndex;
78 
79  // check that we don't leave the valid range of element indices
80  while (m_uiCurrentIndex < m_hashTable->m_uiCapacity)
81  {
82  if (m_hashTable->IsValidEntry(m_uiCurrentIndex))
83  return;
84 
85  ++m_uiCurrentIndex;
86  }
87 
88  // if we fell through this loop, we reached the end of all elements in the container
89  // set the m_uiCurrentCount to maximum, to enable early-out in the future and to make 'IsValid' return 'false'
90  m_uiCurrentCount = m_hashTable->m_uiCount;
91 }

265 template <typename K, typename V, typename H>
266 void ezHashTableBase<K, V, H>::Reserve(ezUInt32 uiCapacity)
267 {
268  const ezUInt64 uiCap64 = static_cast<ezUInt64>(uiCapacity);
269  ezUInt64 uiNewCapacity64 = uiCap64 + (uiCap64 * 2 / 3); // ensure a maximum load of 60%
270 
271  uiNewCapacity64 = ezMath::Min<ezUInt64>(uiNewCapacity64, 0x80000000llu); // the largest power-of-two in 32 bit
272 
273  ezUInt32 uiNewCapacity32 = static_cast<ezUInt32>(uiNewCapacity64 & 0xFFFFFFFF);
274  EZ_ASSERT_DEBUG(uiCapacity <= uiNewCapacity32, "ezHashSet/Map do not support more than 2 billion entries.");
275 
276  if (m_uiCapacity >= uiNewCapacity32)
277  return;
278 
279  uiNewCapacity32 = ezMath::Max<ezUInt32>(ezMath::PowerOfTwo_Ceil(uiNewCapacity32), CAPACITY_ALIGNMENT);
280  SetCapacity(uiNewCapacity32);
281 }

329 template <typename K, typename V, typename H>
330 template <typename CompatibleKeyType, typename CompatibleValueType>
331 bool ezHashTableBase<K, V, H>::Insert(CompatibleKeyType&& key, CompatibleValueType&& value, V* out_oldValue /*= nullptr*/)
332 {
333  Reserve(m_uiCount + 1);
334 
335  ezUInt32 uiIndex = H::Hash(key) & (m_uiCapacity - 1);
336  ezUInt32 uiDeletedIndex = ezInvalidIndex;
337 
338  ezUInt32 uiCounter = 0;
339  while (!IsFreeEntry(uiIndex) && uiCounter < m_uiCapacity)
340  {
341  if (IsDeletedEntry(uiIndex))
342  {
343  if (uiDeletedIndex == ezInvalidIndex)
344  uiDeletedIndex = uiIndex;
345  }
346  else if (H::Equal(m_pEntries[uiIndex].key, key))
347  {
348  if (out_oldValue != nullptr)
349  *out_oldValue = std::move(m_pEntries[uiIndex].value);
350 
351  m_pEntries[uiIndex].value = std::forward<CompatibleValueType>(value); // Either move or copy assignment.
352  return true;
353  }
354  ++uiIndex;
355  if (uiIndex == m_uiCapacity)
356  uiIndex = 0;
357 
358  ++uiCounter;
359  }
360 
361  // new entry
362  uiIndex = uiDeletedIndex != ezInvalidIndex ? uiDeletedIndex : uiIndex;
363 
364  // Both constructions might either be a move or a copy.
365  ezMemoryUtils::CopyOrMoveConstruct(&m_pEntries[uiIndex].key, std::forward<CompatibleKeyType>(key));
366  ezMemoryUtils::CopyOrMoveConstruct(&m_pEntries[uiIndex].value, std::forward<CompatibleValueType>(value));
367 
368  MarkEntryAsValid(uiIndex);
369  ++m_uiCount;
370 
371  return false;
372 }

402 template <typename K, typename V, typename H>
403 void ezHashTableBase<K, V, H>::RemoveInternal(ezUInt32 uiIndex)
404 {
405  ezMemoryUtils::Destruct(&m_pEntries[uiIndex].key, 1);
406  ezMemoryUtils::Destruct(&m_pEntries[uiIndex].value, 1);
407 
408  ezUInt32 uiNextIndex = uiIndex + 1;
409  if (uiNextIndex == m_uiCapacity)
410  uiNextIndex = 0;
411 
412  // if the next entry is free we are at the end of a chain and
413  // can immediately mark this entry as free as well
414  if (IsFreeEntry(uiNextIndex))
415  {
416  MarkEntryAsFree(uiIndex);
417 
418  // run backwards and free all deleted entries in this chain
419  ezUInt32 uiPrevIndex = (uiIndex != 0) ? uiIndex : m_uiCapacity;
420  --uiPrevIndex;
421 
422  while (IsDeletedEntry(uiPrevIndex))
423  {
424  MarkEntryAsFree(uiPrevIndex);
425 
426  if (uiPrevIndex == 0)
427  uiPrevIndex = m_uiCapacity;
428  --uiPrevIndex;
429  }
430  }
431  else
432  {
433  MarkEntryAsDeleted(uiIndex);
434  }
435 
436  --m_uiCount;
437 }

538 V& ezHashTableBase<K, V, H>::FindOrAdd(const K& key, bool* bExisted)
539 {
540  const ezUInt32 uiHash = H::Hash(key);
541  ezUInt32 uiIndex = FindEntry(uiHash, key);
542 
543  if (bExisted)
544  {
545  *bExisted = uiIndex != ezInvalidIndex;
546  }
547 
548  if (uiIndex == ezInvalidIndex)
549  {
550  Reserve(m_uiCount + 1);
551 
552  // search for suitable insertion index again, table might have been resized
553  uiIndex = uiHash & (m_uiCapacity - 1);
554  while (IsValidEntry(uiIndex))
555  {
556  ++uiIndex;
557  if (uiIndex == m_uiCapacity)
558  uiIndex = 0;
559  }
560 
561  // new entry
562  ezMemoryUtils::CopyConstruct(&m_pEntries[uiIndex].key, key, 1);
563  ezMemoryUtils::DefaultConstruct(&m_pEntries[uiIndex].value, 1);
564  MarkEntryAsValid(uiIndex);
565  ++m_uiCount;
566  }
567  return m_pEntries[uiIndex].value;

624 {
625  EZ_ASSERT_DEV(ezMath::IsPowerOf2(uiCapacity), "uiCapacity must be a power of two to avoid modulo during lookup.");

649  EZ_DELETE_RAW_BUFFER(m_pAllocator, pOldEntryFlags);

661 inline ezUInt32 ezHashTableBase<K, V, H>::FindEntry(ezUInt32 uiHash, const CompatibleKeyType& key) const
662 {
663  if (m_uiCapacity > 0)
664  {
665  ezUInt32 uiIndex = uiHash & (m_uiCapacity - 1);
666  ezUInt32 uiCounter = 0;
667  while (!IsFreeEntry(uiIndex) && uiCounter < m_uiCapacity)
668  {
669  if (IsValidEntry(uiIndex) && H::Equal(m_pEntries[uiIndex].key, key))
670  return uiIndex;
671 
672  ++uiIndex;
673  if (uiIndex == m_uiCapacity)
674  uiIndex = 0;
675 
676  ++uiCounter;
677  }
678  }
679  // not found
680  return ezInvalidIndex;
681 }

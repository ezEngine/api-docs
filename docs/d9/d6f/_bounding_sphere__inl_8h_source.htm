194 const ezVec3Template<Type> ezBoundingSphereTemplate<Type>::GetClampedPoint(const ezVec3Template<Type>& vPoint)
195 {
196  const ezVec3Template<Type> vDir = vPoint - m_vCenter;
197  const Type fDistSQR = vDir.GetLengthSquared();
198 
199  // return the point, if it is already inside the sphere
200  if (fDistSQR <= ezMath::Square(m_fRadius))
201  return vPoint;
202 
203  // otherwise return a point on the surface of the sphere
204 
205  const Type fLength = ezMath::Sqrt(fDistSQR);
206 
207  return m_vCenter + m_fRadius * (vDir / fLength);

342  Type* out_fIntersection /* = nullptr */, ezVec3Template<Type>* out_vIntersection /* = nullptr */) const
343 {
344  EZ_ASSERT_DEBUG(vRayDirNormalized.IsNormalized(), "The ray direction must be normalized.");
345 
346  // Ugly Code taken from 'Real Time Rendering First Edition' Page 299
347 
348  const Type fRadiusSQR = ezMath::Square(m_fRadius);
349  const ezVec3Template<Type> vRelPos = m_vCenter - vRayStartPos;
350 
351  const Type d = vRelPos.Dot(vRayDirNormalized);
352  const Type fRelPosLenSQR = vRelPos.GetLengthSquared();
353 
354  if (d < 0.0f && fRelPosLenSQR > fRadiusSQR)
355  return false;
356

358 
359  if (m2 > fRadiusSQR)

363

367  fIntersectionTime = d - q;
368  else
369  fIntersectionTime = d + q;
370 
371  if (out_fIntersection)
372  *out_fIntersection = fIntersectionTime;
373  if (out_vIntersection)
374  *out_vIntersection = vRayStartPos + vRayDirNormalized * fIntersectionTime;

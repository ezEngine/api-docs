206 inline float CalculateSphereScreenCoverage(const ezBoundingSphere& sphere, const ezVec3& vCameraPosition, ezAngle perspectiveCameraFov)
207 {
208 const float fDist = (sphere.m_vCenter - vCameraPosition).GetLength();
209 const float fHalfHeight = ezMath::Tan(perspectiveCameraFov * 0.5f) * fDist;

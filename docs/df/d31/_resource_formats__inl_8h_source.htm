33EZ_FORCE_INLINE bool ezGALResourceFormat::IsIntegerFormat(Enum format)
34{
35 switch (format)
36 {
37 // D16 is actually a 16 bit unorm format
38 case ezGALResourceFormat::D16:
39 // 32bit, 4 channel
40 case ezGALResourceFormat::RGBAUInt:
41 case ezGALResourceFormat::RGBAInt:
42 // 32bit, 3 channel
43 case ezGALResourceFormat::RGBUInt:
44 case ezGALResourceFormat::RGBInt:
45 // 16bit, 4 channel
46 case ezGALResourceFormat::RGBAUShort:
47 case ezGALResourceFormat::RGBAShort:
48 // 16bit, 2 channel
49 case ezGALResourceFormat::RGUInt:
50 case ezGALResourceFormat::RGInt:
51 // packed 32bit, 4 channel
52 case ezGALResourceFormat::RGB10A2UInt:
53 // 8bit, 4 channel
54 case ezGALResourceFormat::RGBAUByte:
55 case ezGALResourceFormat::RGBAByte:
56 // 16bit, 2 channel
57 case ezGALResourceFormat::RGUShort:
58 case ezGALResourceFormat::RGShort:
59 // 8bit, 2 channel
60 case ezGALResourceFormat::RGUByte:
61 case ezGALResourceFormat::RGByte:
62 // 32bit, 1 channel
63 case ezGALResourceFormat::RUInt:
64 case ezGALResourceFormat::RInt:
65 // 16bit, 1 channel
66 case ezGALResourceFormat::RUShort:
67 case ezGALResourceFormat::RShort:

69 case ezGALResourceFormat::RUByte:
70 case ezGALResourceFormat::RByte:
71 return true;

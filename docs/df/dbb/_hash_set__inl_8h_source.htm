144void ezHashSetBase<K, H>::operator=(ezHashSetBase<K, H>&& rhs)
145{
146 // Clear any existing data (calls destructors if necessary)
147 Clear();
148
149 if (m_pAllocator != rhs.m_pAllocator)
150 {
151 Reserve(rhs.m_uiCapacity);
152
153 ezUInt32 uiCopied = 0;
154 for (ezUInt32 i = 0; uiCopied < rhs.GetCount(); ++i)

205template <typename K, typename H>
206void ezHashSetBase<K, H>::Reserve(ezUInt32 uiCapacity)
207{
208 const ezUInt64 uiCap64 = static_cast<ezUInt64>(uiCapacity);
209 ezUInt64 uiNewCapacity64 = uiCap64 + (uiCap64 * 2 / 3); // ensure a maximum load of 60%
210
211 uiNewCapacity64 = ezMath::Min<ezUInt64>(uiNewCapacity64, 0x80000000llu); // the largest power-of-two in 32 bit
212
213 ezUInt32 uiNewCapacity32 = static_cast<ezUInt32>(uiNewCapacity64 & 0xFFFFFFFF);
214 EZ_ASSERT_DEBUG(uiCapacity <= uiNewCapacity32, "ezHashSet/Map do not support more than 2 billion entries.");
215
216 if (m_uiCapacity >= uiNewCapacity32)
217 return;
218
219 uiNewCapacity32 = ezMath::Max<ezUInt32>(ezMath::PowerOfTwo_Ceil(uiNewCapacity32), CAPACITY_ALIGNMENT);
220 SetCapacity(uiNewCapacity32);
221}

268template <typename K, typename H>
269template <typename CompatibleKeyType>
270bool ezHashSetBase<K, H>::Insert(CompatibleKeyType&& key)
271{
272 Reserve(m_uiCount + 1);
273
274 ezUInt32 uiIndex = H::Hash(key) & (m_uiCapacity - 1);
275 ezUInt32 uiDeletedIndex = ezInvalidIndex;
276
277 ezUInt32 uiCounter = 0;
278 while (!IsFreeEntry(uiIndex) && uiCounter < m_uiCapacity)
279 {
280 if (IsDeletedEntry(uiIndex))
281 {
282 if (uiDeletedIndex == ezInvalidIndex)
283 uiDeletedIndex = uiIndex;
284 }
285 else if (H::Equal(m_pEntries[uiIndex], key))
286 {
287 return true;
288 }
289 ++uiIndex;
290 if (uiIndex == m_uiCapacity)
291 uiIndex = 0;
292
293 ++uiCounter;
294 }
295
296 // new entry
297 uiIndex = uiDeletedIndex != ezInvalidIndex ? uiDeletedIndex : uiIndex;
298
299 // Constructions might either be a move or a copy.
300 ezMemoryUtils::CopyOrMoveConstruct(&m_pEntries[uiIndex], std::forward<CompatibleKeyType>(key));
301
302 MarkEntryAsValid(uiIndex);
303 ++m_uiCount;
304
305 return false;
306}

333template <typename K, typename H>
334void ezHashSetBase<K, H>::RemoveInternal(ezUInt32 uiIndex)
335{
336 ezMemoryUtils::Destruct(&m_pEntries[uiIndex], 1);
337
338 ezUInt32 uiNextIndex = uiIndex + 1;
339 if (uiNextIndex == m_uiCapacity)
340 uiNextIndex = 0;
341
342 // if the next entry is free we are at the end of a chain and
343 // can immediately mark this entry as free as well
344 if (IsFreeEntry(uiNextIndex))
345 {
346 MarkEntryAsFree(uiIndex);
347
348 // run backwards and free all deleted entries in this chain
349 ezUInt32 uiPrevIndex = (uiIndex != 0) ? uiIndex : m_uiCapacity;
350 --uiPrevIndex;
351
352 while (IsDeletedEntry(uiPrevIndex))
353 {
354 MarkEntryAsFree(uiPrevIndex);
355
356 if (uiPrevIndex == 0)
357 uiPrevIndex = m_uiCapacity;
358 --uiPrevIndex;
359 }
360 }
361 else
362 {
363 MarkEntryAsDeleted(uiIndex);
364 }
365
366 --m_uiCount;
367}

465template <typename K, typename H>
466void ezHashSetBase<K, H>::SetCapacity(ezUInt32 uiCapacity)
467{
468 EZ_ASSERT_DEBUG(ezMath::IsPowerOf2(uiCapacity), "uiCapacity must be a power of two to avoid modulo during lookup.");
469 const ezUInt32 uiOldCapacity = m_uiCapacity;
470 m_uiCapacity = uiCapacity;
471
472 K* pOldEntries = m_pEntries;
473 ezUInt32* pOldEntryFlags = m_pEntryFlags;
474
475 m_pEntries = EZ_NEW_RAW_BUFFER(m_pAllocator, K, m_uiCapacity);
476 m_pEntryFlags = EZ_NEW_RAW_BUFFER(m_pAllocator, ezUInt32, GetFlagsCapacity());
477 ezMemoryUtils::ZeroFill(m_pEntryFlags, GetFlagsCapacity());
478
479 m_uiCount = 0;
480 for (ezUInt32 i = 0; i < uiOldCapacity; ++i)
481 {
482 if (GetFlags(pOldEntryFlags, i) == VALID_ENTRY)
483 {
484 EZ_VERIFY(!Insert(std::move(pOldEntries[i])), "Implementation error");
485
486 ezMemoryUtils::Destruct(&pOldEntries[i], 1);
487 }
488 }
489
490 EZ_DELETE_RAW_BUFFER(m_pAllocator, pOldEntries);
491 EZ_DELETE_RAW_BUFFER(m_pAllocator, pOldEntryFlags);
492}

501template <typename K, typename H>
502template <typename CompatibleKeyType>
503inline ezUInt32 ezHashSetBase<K, H>::FindEntry(ezUInt32 uiHash, const CompatibleKeyType& key) const
504{
505 if (m_uiCapacity > 0)
506 {
507 ezUInt32 uiIndex = uiHash & (m_uiCapacity - 1);
508 ezUInt32 uiCounter = 0;
509 while (!IsFreeEntry(uiIndex) && uiCounter < m_uiCapacity)
510 {
511 if (IsValidEntry(uiIndex) && H::Equal(m_pEntries[uiIndex], key))
512 return uiIndex;
513
514 ++uiIndex;
515 if (uiIndex == m_uiCapacity)
516 uiIndex = 0;
517
518 ++uiCounter;
519 }
520 }
521 // not found
522 return ezInvalidIndex;
523}

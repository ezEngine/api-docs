87  void WatchNewDirectory(const ezStringBuilder& path, ezDirectoryWatcher::EnumerateChangesFunction& func, bool fireEvent)
88  {
89  ezStringBuilder tmpPath = path;
90  EnsureTrailingSlash(tmpPath);
91 
92  bool directoryAlreadyExists = false;
93  if (m_fileSystemMirror)
94  {
95  m_fileSystemMirror->AddDirectory(tmpPath, &directoryAlreadyExists).AssertSuccess();
96  }
97 
98  if (m_whatToWatch.IsSet(ezDirectoryWatcher::Watch::Creates) && !directoryAlreadyExists && fireEvent)
99  {
100  RemoveTrailingSlash(tmpPath);
101  func(tmpPath, ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::Directory);
102  EnsureTrailingSlash(tmpPath);
103  }
104 
105  if (!m_whatToWatch.IsSet(ezDirectoryWatcher::Watch::Subdirectories))
106  {
107  return;
108  }
109 
110  if (m_pathToWd.Contains(tmpPath))
111  {
112  // Already watching, skip
113  return;
114  }
115 
116  RemoveTrailingSlash(tmpPath);
117  int wd = inotify_add_watch(m_inotifyFd, tmpPath.GetData(), m_inotifyWatchMask);
118  if (wd >= 0)
119  {
120  DEBUG_LOG("Now watching {}", tmpPath);
121  EnsureTrailingSlash(tmpPath);
122  m_pathToWd.Insert(tmpPath, wd);
123  m_wdToPath.Insert(wd, tmpPath);
124 
125  // Whenever we add a directory we might be "to late" to see changes inside it.
126  // So iterate the file system and make sure we track all files / subdirectories
127  ezFileSystemIterator subdirIt;
128 
129  subdirIt.StartSearch(tmpPath.GetData(),
130  m_whatToWatch.IsSet(ezDirectoryWatcher::Watch::Subdirectories)
131  ? ezFileSystemIteratorFlags::ReportFilesAndFoldersRecursive
132  : ezFileSystemIteratorFlags::ReportFiles);
133 
134  ezStringBuilder tmpPath2;
135  for (; subdirIt.IsValid(); subdirIt.Next())
136  {
137  const ezFileStats& stats = subdirIt.GetStats();
138  stats.GetFullPath(tmpPath2);
139  if (stats.m_bIsDirectory)
140  {
141  directoryAlreadyExists = false;
142  if (m_fileSystemMirror)
143  {
144  m_fileSystemMirror->AddDirectory(tmpPath2, &directoryAlreadyExists).AssertSuccess();
145  }
146 
147  if (m_whatToWatch.IsSet(ezDirectoryWatcher::Watch::Creates) && !directoryAlreadyExists)
148  {
149  func(tmpPath2, ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::Directory);
150  }
151 
152  EnsureTrailingSlash(tmpPath2);
153  if (!m_pathToWd.Contains(tmpPath2))
154  {
155  RemoveTrailingSlash(tmpPath2);
156  int wd = inotify_add_watch(m_inotifyFd, tmpPath2.GetData(), m_inotifyWatchMask);
157  if (wd >= 0)
158  {
159  DEBUG_LOG("Now watching {}", tmpPath2);
160  EnsureTrailingSlash(tmpPath2);
161  m_pathToWd.Insert(tmpPath2, wd);
162  m_wdToPath.Insert(wd, tmpPath2);
163  }
164  else
165  {
166  m_pendingDirectories.Insert(tmpPath2);
167  }
168  }
169  }
170  else
171  {
172  bool fileExistsAlready = false;
173  if (m_fileSystemMirror)
174  {
175  m_fileSystemMirror->AddFile(tmpPath2, false, &fileExistsAlready, nullptr).AssertSuccess();
176  }
177  if (m_whatToWatch.IsSet(ezDirectoryWatcher::Watch::Creates) && !fileExistsAlready)
178  {
179  func(tmpPath2, ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::File);
180  }
181  }
182  }
183  }
184  else
185  {
186  m_pendingDirectories.Insert(tmpPath);
187  }
188  }
189 };

204 ezResult ezDirectoryWatcher::OpenDirectory(const ezString& path, ezBitflags<Watch> whatToWatch)
205 {
206  if (m_pImpl->m_inotifyFd >= 0)
207  {
208  return EZ_FAILURE; // already open
209  }
210 
211  m_pImpl->m_inotifyFd = inotify_init();
212  if (m_pImpl->m_inotifyFd < 0)
213  {
214  return EZ_FAILURE; // init failure
215  }
216 
217  // Configure the file descriptor to be non-blocking
218  int flags = fcntl(m_pImpl->m_inotifyFd, F_GETFL, 0);
219  if (fcntl(m_pImpl->m_inotifyFd, F_SETFL, flags | O_NONBLOCK) != 0)
220  {
221  close(m_pImpl->m_inotifyFd);
222  m_pImpl->m_inotifyFd = -1;
223  }
224 
225  ezStringBuilder folder = path;
226  folder.MakeCleanPath();
227  EnsureTrailingSlash(folder);
228 
229  m_pImpl->m_topLevelPath = folder;
230 
231  const int inotifyFd = m_pImpl->m_inotifyFd;
232 
233  uint32_t watchMask = IN_MOVE_SELF;
234  // TODO add IN_MOVE_SELF handling
235 
236 
237  // If we need to watch subdirectories, we always need to be notified if a subdirectory was created or deleted
238  if (whatToWatch.IsSet(Watch::Subdirectories))
239  {
240  watchMask |= IN_CREATE | IN_DELETE | IN_MOVED_FROM | IN_MOVED_TO;
241  }
242  if (whatToWatch.IsSet(Watch::Creates))
243  {
244  watchMask |= IN_CREATE;
245  }
246  if (whatToWatch.IsSet(Watch::Renames))
247  {
248  watchMask |= IN_MOVED_FROM | IN_MOVED_TO;
249  }
250  if (whatToWatch.IsSet(Watch::Writes))
251  {
252  watchMask |= IN_CLOSE_WRITE | IN_CREATE;
253  }
254  if (whatToWatch.IsSet(Watch::Deletes))
255  {
256  watchMask |= IN_DELETE;
257  }
258 
259  int wd = inotify_add_watch(inotifyFd, folder.GetData(), watchMask);
260  if (wd < 0)
261  {
262  close(m_pImpl->m_inotifyFd);
263  m_pImpl->m_inotifyFd = -1;
264  return EZ_FAILURE;
265  }
266 
267  if ((whatToWatch.IsSet(Watch::Deletes) && whatToWatch.IsSet(Watch::Subdirectories)) || whatToWatch.IsSet(Watch::Writes))
268  {
269  // When a sub-folder is moved out of view. We need to trigger delete events for all files inside it.
270  // Thus we need to keep a in memory copy of the file system.
271  m_pImpl->m_fileSystemMirror = EZ_DEFAULT_NEW(ezFileSystemMirrorType);
272  m_pImpl->m_fileSystemMirror->AddDirectory(folder.GetData()).AssertSuccess();
273  }
274 
275  m_pImpl->m_whatToWatch = whatToWatch;
276  m_pImpl->m_inotifyWatchMask = watchMask;
277 
278  DEBUG_LOG("Now watching {}", folder);
279  m_pImpl->m_wdToPath.Insert(wd, folder);
280  m_pImpl->m_pathToWd.Insert(folder, wd);
281 
282  ezFileSystemIterator dirIt;
283  dirIt.StartSearch(folder.GetData(), ezFileSystemIteratorFlags::ReportFoldersRecursive);
284  ezStringBuilder subFolderPath;
285  while (dirIt.IsValid())
286  {
287  dirIt.GetStats().GetFullPath(subFolderPath);
288  wd = inotify_add_watch(inotifyFd, subFolderPath.GetData(), watchMask);
289  if (wd >= 0)
290  {
291  EnsureTrailingSlash(subFolderPath);
292  DEBUG_LOG("Now watching {}", subFolderPath);
293  m_pImpl->m_wdToPath.Insert(wd, subFolderPath);
294  m_pImpl->m_pathToWd.Insert(subFolderPath, wd);
295  }
296  dirIt.Next();
297  }
298 
299  return EZ_SUCCESS;
300 }

320 void ezDirectoryWatcher::EnumerateChanges(EnumerateChangesFunction func, ezTime waitUpTo)
321 {
322  const int inotifyFd = m_pImpl->m_inotifyFd;
323  uint8_t* const buffer = m_pImpl->m_buffer.GetData();
324  const size_t bufferSize = m_pImpl->m_buffer.GetCount();
325 
326  ezStringBuilder tmpPath;
327 
328  const ezBitflags<Watch> whatToWatch = m_pImpl->m_whatToWatch;
329  ezFileSystemMirrorType* mirror = m_pImpl->m_fileSystemMirror.Borrow();
330 
331  MoveEvent lastMoveFrom;
332 
333  // An orpahend move from is when we see a move from, but no move to
334  // This means the file was moved outside of our view of the file system
335  // tread this as a delete.
336  auto processOrphanedMoveFrom = [&](MoveEvent& moveFrom) {
337  if (whatToWatch.IsSet(Watch::Deletes))
338  {
339  if (!moveFrom.isDirectory)
340  {
341  func(moveFrom.path.GetData(), ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::File);
342  mirror->RemoveFile(moveFrom.path).AssertSuccess();
343  m_pImpl->m_pendingDirectories.Remove(moveFrom.path);
344  }
345  else
346  {
347  ezStringBuilder dirPath;
348  mirror->Enumerate(moveFrom.path, [&](const char* path, typename ezFileSystemMirrorType::Type type) {
349  if (type == ezFileSystemMirrorType::Type::File)
350  {
351  func(path, ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::File);
352  }
353  else
354  {
355  func(path, ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::Directory);
356  dirPath = path;
357  EnsureTrailingSlash(dirPath);
358  auto it = m_pImpl->m_pathToWd.Find(dirPath);
359  if (it.IsValid())
360  {
361  DEBUG_LOG("No longer watching {}", it.Key());
362  inotify_rm_watch(inotifyFd, it.Value());
363  m_pImpl->m_wdToPath.Remove(it.Value());
364  m_pImpl->m_pathToWd.Remove(it);
365  }
366  }
367  })
368  .AssertSuccess();
369  mirror->RemoveDirectory(moveFrom.path).AssertSuccess();
370 
371  func(moveFrom.path, ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::Directory);
372 
373  dirPath = moveFrom.path;
374  EnsureTrailingSlash(dirPath);
375  auto it = m_pImpl->m_pathToWd.Find(dirPath);
376  EZ_ASSERT_DEBUG(it.IsValid(), "path should exist");
377  if (it.IsValid())
378  {
379  DEBUG_LOG("No longer watching {}", it.Key());
380  m_pImpl->m_wdToPath.Remove(it.Value());
381  m_pImpl->m_pathToWd.Remove(it);
382  }
383  }
384  }
385  moveFrom.Clear();
386  };
387 
388  if (inotifyFd >= 0)
389  {
390  int timeout = static_cast<int>(waitUpTo.GetMilliseconds());
391  if (timeout > 0)
392  {
393  struct pollfd pollFor = {inotifyFd, POLLIN, 0};
394  int pollResult = poll(&pollFor, 1, timeout);
395  if (pollResult < 0)
396  {
397  // Error, stop
398  ezLog::Error("Unexpected result from poll when watching directory {}", m_sDirectoryPath);
399  CloseDirectory();
400  return;
401  }
402  else if (pollResult == 0)
403  {
404  return; // timeout, no results
405  }
406  }
407 
408  ssize_t numBytesRead = 0;
409  while ((numBytesRead = read(inotifyFd, buffer, bufferSize)) > 0)
410  {
411  size_t curPos = 0;
412  while (curPos < numBytesRead)
413  {
414  const struct inotify_event* event = (struct inotify_event*)(buffer + curPos);
415 
416  auto it = m_pImpl->m_wdToPath.Find(event->wd);
417  if (it.IsValid() && event->len > 0)
418  {
419  tmpPath = it.Value();
420  tmpPath.AppendPath(event->name);
421 
422  const char* type = "file";
423  if (IsDirectory(event->mask))
424  {
425  type = "folder";
426  }
427 
428  if (event->mask & IN_CREATE)
429  {
430  DEBUG_LOG("IN_CREATE {} {} {} {}", type, tmpPath, event->cookie, event->mask);
431 
432  if (IsDirectory(event->mask))
433  {
434  m_pImpl->WatchNewDirectory(tmpPath, func, true);
435  }
436  else
437  {
438  bool fileExistsAlready = false;
439  if (mirror != nullptr)
440  {
441  mirror->AddFile(tmpPath, true, &fileExistsAlready, nullptr).AssertSuccess();
442  }
443 
444  if (whatToWatch.IsSet(Watch::Creates) && !fileExistsAlready)
445  {
446  func(tmpPath.GetData(), ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::File);
447  }
448  }
449  }
450  else if (event->mask & IN_CLOSE_WRITE)
451  {
452  DEBUG_LOG("IN_CLOSE_WRITE {} {} {} {}", type, tmpPath, event->cookie, event->mask);
453  if (whatToWatch.IsSet(Watch::Writes) && IsFile(event->mask))
454  {
455  bool addPending = false;
456  if (mirror)
457  {
458  mirror->AddFile(tmpPath, false, nullptr, &addPending).AssertSuccess();
459  }
460  if (!addPending)
461  {
462  func(tmpPath.GetData(), ezDirectoryWatcherAction::Modified, ezDirectoryWatcherType::File);
463  }
464  }
465  }
466  else if (event->mask & IN_DELETE)
467  {
468  DEBUG_LOG("IN_DELETE {} {} {}", type, tmpPath, event->cookie);
469 
470  if (IsDirectory(event->mask))
471  {
472  m_pImpl->m_pendingDirectories.Remove(tmpPath);
473 
474  if (mirror)
475  {
476  mirror->RemoveDirectory(tmpPath).AssertSuccess();
477  }
478 
479  if (whatToWatch.IsSet(Watch::Deletes))
480  {
481  func(tmpPath.GetData(), ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::Directory);
482  }
483 
484  if (whatToWatch.IsSet(Watch::Subdirectories))
485  {
486  EnsureTrailingSlash(tmpPath);
487  auto deletedDirIt = m_pImpl->m_pathToWd.Find(tmpPath);
488  if (deletedDirIt.IsValid())
489  {
490  int deletedWd = deletedDirIt.Value();
491  deletedDirIt = m_pImpl->m_pathToWd.Remove(deletedDirIt);
492  inotify_rm_watch(inotifyFd, deletedWd);
493  m_pImpl->m_wdToPath.Remove(deletedWd);
494  DEBUG_LOG("No longer watching {}", tmpPath);
495  }
496  }
497  }
498  else
499  {
500  if (mirror)
501  {
502  mirror->RemoveFile(tmpPath).AssertSuccess();
503  }
504 
505  if (whatToWatch.IsSet(Watch::Deletes))
506  {
507  func(tmpPath.GetData(), ezDirectoryWatcherAction::Removed, ezDirectoryWatcherType::File);
508  }
509  }
510  }
511  else if (event->mask & IN_MOVED_FROM)
512  {
513  DEBUG_LOG("IN_MOVED_FROM {} {} {}", type, tmpPath, event->cookie);
514 
515  if (!lastMoveFrom.IsEmpty())
516  {
517  processOrphanedMoveFrom(lastMoveFrom);
518  }
519 
520  lastMoveFrom.path = tmpPath;
521  lastMoveFrom.cookie = event->cookie;
522  lastMoveFrom.isDirectory = IsDirectory(event->mask);
523  }
524  else if (event->mask & IN_MOVED_TO)
525  {
526  DEBUG_LOG("IN_MOVED_TO {} {} {}", type, tmpPath, event->cookie);
527 
528  if (!lastMoveFrom.IsEmpty() && lastMoveFrom.cookie != event->cookie)
529  {
530  // orphaned move from
531  processOrphanedMoveFrom(lastMoveFrom);
532  }
533 
534  if (lastMoveFrom.IsEmpty())
535  {
536  // Orphaned move to, treat as add
537  if (IsFile(event->mask))
538  {
539  bool fileAlreadyExists = false;
540  if (mirror)
541  {
542  mirror->AddFile(tmpPath, false, &fileAlreadyExists, nullptr).AssertSuccess();
543  }
544 
545  if (whatToWatch.IsSet(Watch::Creates) && !fileAlreadyExists)
546  {
547  func(tmpPath.GetData(), ezDirectoryWatcherAction::Added, ezDirectoryWatcherType::File);
548  }
549  }
550  else
551  {
552  m_pImpl->WatchNewDirectory(tmpPath, func, true);
553  }
554  }
555  else
556  {
557  // regular move
558  if (whatToWatch.IsSet(Watch::Renames))
559  {
560  func(lastMoveFrom.path.GetData(), ezDirectoryWatcherAction::RenamedOldName, IsFile(event->mask) ? ezDirectoryWatcherType::File : ezDirectoryWatcherType::Directory);
561  func(tmpPath.GetData(), ezDirectoryWatcherAction::RenamedNewName, IsFile(event->mask) ? ezDirectoryWatcherType::File : ezDirectoryWatcherType::Directory);
562  }
563 
564  if (IsDirectory(event->mask))
565  {
566  if (m_pImpl->m_pendingDirectories.Contains(lastMoveFrom.path))
567  {
568  m_pImpl->m_pendingDirectories.Remove(lastMoveFrom.path);
569  m_pImpl->WatchNewDirectory(tmpPath, func, false);
570  }
571  else
572  {
573  ezStringBuilder moveFromPath = lastMoveFrom.path;
574  EnsureTrailingSlash(moveFromPath);
575  EnsureTrailingSlash(tmpPath);
576 
577  ezDynamicArray<RenamedDirectory> renamedDirectories;
578  for (auto it = m_pImpl->m_pathToWd.GetIterator(); it.IsValid();)
579  {
580  if (it.Key().StartsWith(moveFromPath))
581  {
582  m_pImpl->m_wdToPath.Remove(it.Value());
583  ezStringBuilder fixedPath = it.Key();
584  fixedPath.Shrink(moveFromPath.GetCharacterCount(), 0);
585  fixedPath.Prepend(tmpPath);
586  renamedDirectories.PushBack({fixedPath, it.Value()});
587  it = m_pImpl->m_pathToWd.Remove(it);
588  }
589  else
590  {
591  it.Next();
592  }
593  }
594 
595  for (auto& renamedDirectory : renamedDirectories)
596  {
597  m_pImpl->m_pathToWd.Insert(renamedDirectory.path, renamedDirectory.wd);
598  m_pImpl->m_wdToPath.Insert(renamedDirectory.wd, std::move(renamedDirectory.path));
599  }
600  }
601  }
602 
603  if (mirror)
604  {
605  if (IsFile(event->mask))
606  {
607  mirror->RemoveFile(lastMoveFrom.path).AssertSuccess();
608  mirror->AddFile(tmpPath, false, nullptr, nullptr).AssertSuccess();
609  }
610  else
611  {
612  mirror->MoveDirectory(lastMoveFrom.path, tmpPath).AssertSuccess();
613  }
614  }
615  lastMoveFrom.Clear();
616  }
617  }
618  }
619  else
620  {
621  DEBUG_LOG("UNKNOWN {}", event->mask);
622  }
623 
624  curPos += sizeof(struct inotify_event) + event->len;
625  }
626  }
627  }
628 
629  if (!lastMoveFrom.IsEmpty())
630  {
631  processOrphanedMoveFrom(lastMoveFrom);
632  }
633 }

635 void ezDirectoryWatcher::EnumerateChanges(ezArrayPtr<ezDirectoryWatcher*> watchers, EnumerateChangesFunction func, ezTime waitUpTo)
636 {
637  int timeout = static_cast<int>(waitUpTo.GetMilliseconds());
638  if (timeout > 0)
639  {
640  ezHybridArray<struct pollfd, 16> pollFor;
641  pollFor.SetCount(watchers.GetCount());
642 
643  for (ezUInt32 i = 0; i < watchers.GetCount(); ++i)
644  {
645  ezDirectoryWatcher* curWatcher = watchers[i];
646  pollFor[i] = {curWatcher->m_pImpl->m_inotifyFd, POLLIN, 0};
647  }
648 
649 
650  int pollResult = poll(pollFor.GetData(), pollFor.GetCount(), timeout);
651  if (pollResult < 0)
652  {
653  // Error, stop
654  ezLog::Error("Unexpected result from poll enumerating multiple watchers");
655  return;
656  }
657  else if (pollResult == 0)
658  {
659  return; // timeout, no results
660  }
661  }
662 
663  for (ezDirectoryWatcher* watcher : watchers)
664  {
665  watcher->EnumerateChanges(func);
666  }
667 }

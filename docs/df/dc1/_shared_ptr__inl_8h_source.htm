290template <typename T>
291EZ_ALWAYS_INLINE void ezSharedPtr<T>::ReleaseReferenceIfValid()
292{
293 if (m_pInstance != nullptr)
294 {
295 if (m_pInstance->ReleaseRef() == 0)
296 {
297 auto pNonConstInstance = const_cast<typename ezTypeTraits<T>::NonConstType*>(m_pInstance);
298 EZ_ASSERT_DEV(m_pAllocator != nullptr, "Fake shared pointers should never be released");
299 EZ_DELETE(m_pAllocator, pNonConstInstance);
300 }
301
302 m_pInstance = nullptr;
303 m_pAllocator = nullptr;
304 }

138EZ_ALWAYS_INLINE float ezSpline::ClampAndSplitT(float fT, ezUInt32& out_uiIndex) const
139{
140 float fNumPoints = static_cast<float>(m_ControlPoints.GetCount());
141 if (m_bClosed)
142 {
143 fT = (fT < 0.0f || fT >= fNumPoints) ? 0.0f : fT;
144 }
145 else
146 {
147 fT = ezMath::Clamp(fT, 0.0f, fNumPoints - 1.0f);
148 }
149
150 const float fIndex = ezMath::Floor(fT);
151 out_uiIndex = static_cast<ezUInt32>(fIndex);
152 return fT - fIndex;
153}
